# 10.1 前端基础

## 一、HTML 基础

### 1.1 HTML 结构

```html
<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <title>页面标题</title>
</head>
<body>
    <!-- 页面内容 -->
</body>
</html>
```

### 1.2 常用标签

```html
<!-- 标题 -->
<h1>一级标题</h1>
<h2>二级标题</h2>

<!-- 段落 -->
<p>这是一个段落</p>

<!-- 链接 -->
<a href="https://example.com">链接</a>

<!-- 图片 -->
<img src="image.jpg" alt="描述">

<!-- 列表 -->
<ul>
    <li>无序列表项 1</li>
    <li>无序列表项 2</li>
</ul>

<ol>
    <li>有序列表项 1</li>
    <li>有序列表项 2</li>
</ol>

<!-- 表格 -->
<table>
    <tr>
        <th>表头 1</th>
        <th>表头 2</th>
    </tr>
    <tr>
        <td>单元格 1</td>
        <td>单元格 2</td>
    </tr>
</table>

<!-- 表单 -->
<form action="/submit" method="post">
    <input type="text" name="username" placeholder="用户名">
    <input type="password" name="password" placeholder="密码">
    <input type="email" name="email" placeholder="邮箱">
    <input type="submit" value="提交">
</form>

<!-- 分区 -->
<div>块级元素</div>
<span>行内元素</span>
```

## 二、CSS 基础

### 2.1 引入方式

```html
<!-- 内联样式 -->
<p style="color: red; font-size: 16px;">文本</p>

<!-- 内部样式 -->
<style>
    p { color: red; }
</style>

<!-- 外部样式 -->
<link rel="stylesheet" href="style.css">
```

### 2.2 选择器

```css
/* 标签选择器 */
p { color: red; }

/* 类选择器 */
.class-name { color: blue; }

/* ID 选择器 */
#id-name { color: green; }

/* 后代选择器 */
div p { color: red; }

/* 子元素选择器 */
div > p { color: red; }

/* 伪类选择器 */
a:hover { color: red; }
li:first-child { font-weight: bold; }
```

### 2.3 盒模型

```css
.box {
    width: 200px;           /* 内容宽度 */
    height: 100px;          /* 内容高度 */
    padding: 10px;          /* 内边距 */
    border: 1px solid #000; /* 边框 */
    margin: 20px;           /* 外边距 */
}

/* 盒模型类型 */
.box {
    box-sizing: content-box;  /* 标准盒模型（默认）*/
    /* box-sizing: border-box;  /* IE 盒模型 */
}
```

### 2.4 布局

```css
/* Flex 布局 */
.container {
    display: flex;
    justify-content: center;  /* 水平居中 */
    align-items: center;      /* 垂直居中 */
}

/* Grid 布局 */
.container {
    display: grid;
    grid-template-columns: repeat(3, 1fr);
    gap: 10px;
}

/* 定位 */
.relative { position: relative; }
.absolute { position: absolute; }
.fixed { position: fixed; }
.sticky { position: sticky; }
```

## 三、JavaScript 基础

### 3.1 变量声明

```javascript
// var（已不推荐）
var name = "张三";

// let（块级作用域）
let age = 18;

// const（常量）
const PI = 3.14159;
```

### 3.2 数据类型

```javascript
// 基本类型
let str = "Hello";       // 字符串
let num = 123;           // 数字
let bool = true;         // 布尔
let nothing = null;      // 空
let undef;               // 未定义

// 引用类型
let arr = [1, 2, 3];     // 数组
let obj = {              // 对象
    name: "张三",
    age: 18
};
```

### 3.3 函数

```javascript
// 函数声明
function add(a, b) {
    return a + b;
}

// 函数表达式
const subtract = function(a, b) {
    return a - b;
};

// 箭头函数（ES6）
const multiply = (a, b) => a * b;

// 默认参数
function greet(name = "World") {
    return "Hello, " + name;
}
```

### 3.4 DOM 操作

```javascript
// 获取元素
const elem = document.getElementById("myId");
const elems = document.getElementsByClassName("myClass");
const elem = document.querySelector(".myClass");
const elems = document.querySelectorAll("p");

// 修改内容
elem.innerHTML = "<strong>内容</strong>";
elem.textContent = "纯文本";

// 修改样式
elem.style.color = "red";
elem.style.fontSize = "16px";

// 添加/删除类
elem.classList.add("active");
elem.classList.remove("active");
elem.classList.toggle("active");

// 事件监听
elem.addEventListener("click", function(e) {
    console.log("点击了", e.target);
});
```

### 3.5 BOM 操作

```javascript
// 窗口操作
window.alert("警告");
window.confirm("确认吗？");
window.prompt("请输入");

// 定时器
setTimeout(() => {
    console.log("1 秒后执行");
}, 1000);

setInterval(() => {
    console.log("每秒执行");
}, 1000);

// 清除定时器
clearTimeout(timerId);
clearInterval(timerId);

// localStorage
localStorage.setItem("key", "value");
const value = localStorage.getItem("key");
localStorage.removeItem("key");
localStorage.clear();
```

## 四、AJAX

### 4.1 XMLHttpRequest

```javascript
const xhr = new XMLHttpRequest();

xhr.open("GET", "/api/users", true);

xhr.onload = function() {
    if (xhr.status === 200) {
        const data = JSON.parse(xhr.responseText);
        console.log(data);
    }
};

xhr.onerror = function() {
    console.error("请求失败");
};

xhr.send();
```

### 4.2 Fetch API

```javascript
// GET 请求
fetch("/api/users")
    .then(response => response.json())
    .then(data => console.log(data))
    .catch(error => console.error(error));

// POST 请求
fetch("/api/users", {
    method: "POST",
    headers: {
        "Content-Type": "application/json"
    },
    body: JSON.stringify({
        name: "张三",
        age: 18
    })
})
    .then(response => response.json())
    .then(data => console.log(data))
    .catch(error => console.error(error));

// async/await
async function getUsers() {
    try {
        const response = await fetch("/api/users");
        const data = await response.json();
        console.log(data);
    } catch (error) {
        console.error(error);
    }
}
```

## 六、综合示例

### 6.1 登录页面

```html
<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <title>登录</title>
    <style>
        .form-group { margin: 10px 0; }
        .form-group label { display: inline-block; width: 80px; }
        .form-group input { padding: 5px; width: 200px; }
        .btn { padding: 10px 20px; background: #007bff; color: white; border: none; cursor: pointer; }
        .error { color: red; }
    </style>
</head>
<body>
    <h1>用户登录</h1>
    <form id="loginForm">
        <div class="form-group">
            <label>用户名：</label>
            <input type="text" id="username" required>
        </div>
        <div class="form-group">
            <label>密码：</label>
            <input type="password" id="password" required>
        </div>
        <div class="form-group">
            <button type="submit" class="btn">登录</button>
        </div>
        <div id="error" class="error"></div>
    </form>

    <script>
        document.getElementById("loginForm").addEventListener("submit", async function(e) {
            e.preventDefault();
            
            const username = document.getElementById("username").value;
            const password = document.getElementById("password").value;
            
            try {
                const response = await fetch("/api/login", {
                    method: "POST",
                    headers: { "Content-Type": "application/json" },
                    body: JSON.stringify({ username, password })
                });
                
                const result = await response.json();
                
                if (result.code === 200) {
                    localStorage.setItem("token", result.token);
                    window.location.href = "/index.html";
                } else {
                    document.getElementById("error").textContent = result.message;
                }
            } catch (error) {
                document.getElementById("error").textContent = "网络错误";
            }
        });
    </script>
</body>
</html>
```

### 6.2 用户列表

```html
<!DOCTYPE html>
<html>
<head>
    <meta charset="UTF-8">
    <title>用户列表</title>
    <style>
        table { border-collapse: collapse; width: 100%; }
        th, td { border: 1px solid #ddd; padding: 8px; text-align: left; }
        th { background: #f4f4f4; }
        tr:hover { background: #f5f5f5; }
    </style>
</head>
<body>
    <h1>用户列表</h1>
    <table>
        <thead>
            <tr>
                <th>ID</th>
                <th>用户名</th>
                <th>邮箱</th>
                <th>年龄</th>
                <th>操作</th>
            </tr>
        </thead>
        <tbody id="userTable">
        </tbody>
    </table>

    <script>
        async function loadUsers() {
            try {
                const response = await fetch("/api/users");
                const users = await response.json();
                
                const tbody = document.getElementById("userTable");
                tbody.innerHTML = users.map(user => `
                    <tr>
                        <td>${user.id}</td>
                        <td>${user.username}</td>
                        <td>${user.email}</td>
                        <td>${user.age}</td>
                        <td>
                            <button onclick="editUser(${user.id})">编辑</button>
                            <button onclick="deleteUser(${user.id})">删除</button>
                        </td>
                    </tr>
                `).join("");
            } catch (error) {
                console.error("加载失败", error);
            }
        }

        async function deleteUser(id) {
            if (!confirm("确定删除？")) return;
            
            await fetch(`/api/users/${id}`, { method: "DELETE" });
            loadUsers();
        }

        function editUser(id) {
            window.location.href = `/edit.html?id=${id}`;
        }

        loadUsers();
    </script>
</body>
</html>
```

## 七、小结

1. **HTML**：页面结构、常用标签、表单
2. **CSS**：选择器、盒模型、布局（Flex、Grid）
3. **JavaScript**：变量、函数、DOM 操作、事件
4. **AJAX**：XMLHttpRequest、Fetch API
5. **JSON**：数据格式、序列化/反序列化
