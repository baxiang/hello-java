# 14.5 分布式外卖平台

## 一、项目介绍

### 1.1 项目概述

分布式外卖平台是一个基于 Spring Cloud 微服务架构的外卖订餐系统，包含用户端、商家端、骑手端和管理后台。

**技术栈**：
- Spring Boot 3.4
- Spring Cloud Alibaba 2023.x
- Nacos 2.x（注册中心 + 配置中心）
- OpenFeign（服务调用）
- Sentinel 1.8（熔断限流）
- Gateway 4.x（API 网关）
- Seata 1.7（分布式事务）
- MySQL 8.0 + Redis 7.0 + MongoDB 7.0
- Docker + Kubernetes

### 2.1 用户服务

```java
@Service
public class UserServiceImpl extends ServiceImpl<UserMapper, User> implements UserService {
    
    @Autowired
    private PasswordEncoder passwordEncoder;
    
    @Autowired
    private JwtUtil jwtUtil;
    
    /**
     * 用户注册
     */
    @Override
    public Result<UserVO> register(RegisterDTO dto) {
        // 检查手机号是否已注册
        User existUser = lambdaQuery()
            .eq(User::getPhone, dto.getPhone())
            .one();
        
        if (existUser != null) {
            return Result.error("手机号已注册");
        }
        
        // 创建用户
        User user = new User();
        user.setPhone(dto.getPhone());
        user.setPassword(passwordEncoder.encode(dto.getPassword()));
        user.setNickname(dto.getNickname());
        user.setAvatar(dto.getAvatar());
        
        baseMapper.insert(user);
        
        return Result.success(convertToVO(user));
    }
    
    /**
     * 用户登录
     */
    @Override
    public Result<LoginVO> login(LoginDTO dto) {
        // 查询用户
        User user = lambdaQuery()
            .eq(User::getPhone, dto.getPhone())
            .one();
        
        if (user == null) {
            return Result.error("用户不存在");
        }
        
        // 验证密码
        if (!passwordEncoder.matches(dto.getPassword(), user.getPassword())) {
            return Result.error("密码错误");
        }
        
        // 生成 Token
        String token = jwtUtil.generateToken(user.getId(), user.getPhone());
        
        LoginVO vo = new LoginVO();
        vo.setToken(token);
        vo.setUser(convertToVO(user));
        
        return Result.success(vo);
    }
}
```

### 2.2 商家服务

```java
@Service
public class ShopServiceImpl extends ServiceImpl<ShopMapper, Shop> implements ShopService {
    
    /**
     * 获取商家详情
     */
    @Override
    public Result<ShopDetailVO> getShopDetail(Long id) {
        Shop shop = baseMapper.selectById(id);
        if (shop == null) {
            return Result.error("商家不存在");
        }
        
        ShopDetailVO vo = new ShopDetailVO();
        BeanUtils.copyProperties(shop, vo);
        
        // 查询商家评分
        Double score = shopMapper.selectAverageScore(id);
        vo.setScore(score != null ? score : 0.0);
        
        // 查询商家公告
        List<ShopAnnouncement> announcements = shopAnnouncementMapper.selectList(
            new LambdaQueryWrapper<ShopAnnouncement>()
                .eq(ShopAnnouncement::getShopId, id)
                .eq(ShopAnnouncement::getStatus, 1)
                .orderByDesc(ShopAnnouncement::getCreateTime)
        );
        vo.setAnnouncements(announcements);
        
        // 查询商家营业时间
        ShopWorkingTime workingTime = shopWorkingTimeMapper.selectOne(
            new LambdaQueryWrapper<ShopWorkingTime>()
                .eq(ShopWorkingTime::getShopId, id)
        );
        vo.setWorkingTime(workingTime);
        
        return Result.success(vo);
    }
    
    /**
     * 搜索商家
     */
    @Override
    public Result<IPage<ShopVO>> searchShops(ShopSearchDTO dto) {
        Page<Shop> page = new Page<>(dto.getPage(), dto.getSize());
        
        LambdaQueryWrapper<Shop> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(Shop::getStatus, 1);  // 只查询营业中
        
        if (dto.getLatitude() != null && dto.getLongitude() != null) {
            // 按距离排序（使用 MySQL 空间函数或 Redis GEO）
            // 这里简化处理
        }
        
        if (StringUtils.hasText(dto.getKeyword())) {
            wrapper.and(w -> w
                .like(Shop::getName, dto.getKeyword())
                .or()
                .like(Shop::getAddress, dto.getKeyword())
            );
        }
        
        if (dto.getSortType() != null) {
            switch (dto.getSortType()) {
                case 1: wrapper.orderByDesc(Shop::getSales); break;  // 销量
                case 2: wrapper.orderByDesc(Shop::getScore); break;  // 评分
                case 3: wrapper.orderByAsc(Shop::getDistance); break; // 距离
            }
        }
        
        IPage<Shop> result = baseMapper.selectPage(page, wrapper);
        
        IPage<ShopVO> voPage = result.convert(this::convertToVO);
        
        return Result.success(voPage);
    }
}
```

### 2.4 订单服务（分布式事务）

```java
@Service
public class OrderServiceImpl extends ServiceImpl<OrderMapper, Order> implements OrderService {
    
    @Autowired
    private DishClient dishClient;
    
    @Autowired
    private ShopClient shopClient;
    
    @Autowired
    private IdWorker idWorker;
    
    /**
     * 创建订单（分布式事务）
     */
    @Override
    @GlobalTransactional
    public Result<OrderVO> createOrder(Long userId, CreateOrderDTO dto) {
        // 1. 验证商家营业状态
        Result<ShopVO> shopResult = shopClient.getById(dto.getShopId());
        if (shopResult.getCode() != 200 || shopResult.getData() == null) {
            return Result.error("商家不存在");
        }
        
        // 2. 验证菜品信息
        Result<List<DishVO>> dishResult = dishClient.getByShopId(dto.getShopId());
        if (dishResult.getCode() != 200) {
            return Result.error("菜品信息获取失败");
        }
        
        Map<Long, DishVO> dishMap = dishResult.getData().stream()
            .collect(Collectors.toMap(DishVO::getId, Function.identity()));
        
        // 3. 计算订单金额
        BigDecimal totalAmount = BigDecimal.ZERO;
        List<OrderItem> items = new ArrayList<>();
        
        for (OrderItemDTO itemDTO : dto.getItems()) {
            DishVO dish = dishMap.get(itemDTO.getDishId());
            if (dish == null) {
                throw new BusinessException("菜品不存在：" + itemDTO.getDishId());
            }
            
            OrderItem item = new OrderItem();
            item.setDishId(dish.getId());
            item.setDishName(dish.getName());
            item.setPrice(dish.getPrice());
            item.setQuantity(itemDTO.getQuantity());
            item.setTotalPrice(dish.getPrice().multiply(new BigDecimal(itemDTO.getQuantity())));
            
            items.add(item);
            totalAmount = totalAmount.add(item.getTotalPrice());
        }
        
        // 4. 创建订单
        String orderNo = "ORD" + idWorker.nextId();
        Order order = new Order();
        order.setOrderNo(orderNo);
        order.setUserId(userId);
        order.setShopId(dto.getShopId());
        order.setTotalAmount(totalAmount);
        order.setPayAmount(totalAmount);
        order.setStatus(OrderStatus.PENDING_PAYMENT);
        order.setAddress(dto.getAddress());
        order.setRemark(dto.getRemark());
        
        baseMapper.insert(order);
        
        // 5. 创建订单项
        for (OrderItem item : items) {
            item.setOrderId(order.getId());
        }
        orderItemService.saveBatch(items);
        
        // 6. 转换为 VO
        OrderVO vo = convertToVO(order);
        vo.setItems(items.stream().map(this::convertItemToVO).collect(Collectors.toList()));
        
        return Result.success(vo);
    }
}
```

### 2.5 骑手服务

```java
@Service
public class RiderServiceImpl extends ServiceImpl<RiderMapper, Rider> implements RiderService {
    
    @Autowired
    private OrderClient orderClient;
    
    /**
     * 骑手接单
     */
    @Override
    @Transactional
    public Result<Void> acceptOrder(Long riderId, Long orderId) {
        // 检查订单状态
        Result<OrderVO> orderResult = orderClient.getById(orderId);
        if (orderResult.getCode() != 200) {
            return Result.error("订单不存在");
        }
        
        OrderVO order = orderResult.getData();
        if (!OrderStatus.PENDING_DELIVERY.equals(order.getStatus())) {
            return Result.error("订单状态不允许接单");
        }
        
        // 检查骑手状态
        Rider rider = baseMapper.selectById(riderId);
        if (rider == null || rider.getStatus() != RiderStatus.AVAILABLE) {
            return Result.error("骑手不可用");
        }
        
        // 更新订单骑手
        orderClient.updateRider(orderId, riderId);
        
        // 更新骑手状态
        rider.setStatus(RiderStatus.BUSY);
        rider.setCurrentOrderId(orderId);
        baseMapper.updateById(rider);
        
        return Result.success();
    }
    
    /**
     * 更新配送状态
     */
    @Override
    public Result<Void> updateDeliveryStatus(Long riderId, Long orderId, Integer status) {
        // 验证骑手权限
        Rider rider = baseMapper.selectById(riderId);
        if (!rider.getCurrentOrderId().equals(orderId)) {
            return Result.error("无权操作此订单");
        }
        
        // 更新订单状态
        orderClient.updateStatus(orderId, status);
        
        // 如果配送完成，更新骑手状态
        if (DeliveryStatus.COMPLETED.equals(status)) {
            rider.setStatus(RiderStatus.AVAILABLE);
            rider.setCurrentOrderId(null);
            baseMapper.updateById(rider);
        }
        
        return Result.success();
    }
}
```

### 2.6 支付服务

```java
@Service
public class PaymentServiceImpl implements PaymentService {
    
    @Autowired
    private OrderClient orderClient;
    
    @Autowired
    private AlipayService alipayService;
    
    @Autowired
    private WechatPayService wechatPayService;
    
    /**
     * 创建支付
     */
    @Override
    public Result<PaymentVO> createPayment(Long userId, Long orderId, String channel) {
        // 获取订单信息
        Result<OrderVO> orderResult = orderClient.getById(orderId);
        if (orderResult.getCode() != 200) {
            return Result.error("订单不存在");
        }
        
        OrderVO order = orderResult.getData();
        
        // 验证订单状态
        if (!OrderStatus.PENDING_PAYMENT.equals(order.getStatus())) {
            return Result.error("订单状态不允许支付");
        }
        
        // 创建支付记录
        Payment payment = new Payment();
        payment.setUserId(userId);
        payment.setOrderId(orderId);
        payment.setOrderNo(order.getOrderNo());
        payment.setAmount(order.getPayAmount());
        payment.setChannel(channel);
        payment.setStatus(PaymentStatus.PENDING);
        
        paymentMapper.insert(payment);
        
        PaymentVO vo = new PaymentVO();
        vo.setPaymentId(payment.getId());
        
        // 调用支付渠道
        if ("alipay".equals(channel)) {
            String alipayUrl = alipayService.createOrder(order.getOrderNo(), order.getPayAmount());
            vo.setPayUrl(alipayUrl);
        } else if ("wechat".equals(channel)) {
            Map<String, String> wechatParams = wechatPayService.createOrder(order.getOrderNo(), order.getPayAmount());
            vo.setPayParams(wechatParams);
        }
        
        return Result.success(vo);
    }
    
    /**
     * 支付回调
     */
    @Override
    @Transactional
    public Result<Void> callback(String orderNo, String channel, Map<String, String> params) {
        // 验证签名
        boolean verified = false;
        if ("alipay".equals(channel)) {
            verified = alipayService.verify(params);
        } else if ("wechat".equals(channel)) {
            verified = wechatPayService.verify(params);
        }
        
        if (!verified) {
            return Result.error("签名验证失败");
        }
        
        // 更新支付状态
        Payment payment = paymentMapper.selectOne(
            new LambdaQueryWrapper<Payment>()
                .eq(Payment::getOrderNo, orderNo)
                .eq(Payment::getChannel, channel)
        );
        
        if (payment == null) {
            return Result.error("支付记录不存在");
        }
        
        payment.setStatus(PaymentStatus.SUCCESS);
        payment.setTransactionId(params.get("trade_no"));
        payment.setPayTime(LocalDateTime.now());
        paymentMapper.updateById(payment);
        
        // 更新订单状态
        orderClient.paySuccess(orderNo);
        
        return Result.success();
    }
}
```
