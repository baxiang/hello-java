# 14.2 在线商城系统

## 一、项目介绍

### 1.1 项目概述

在线商城系统是一个 B2C 电商平台，包含商品管理、购物车、订单、支付等功能。

**技术栈**：
- Spring Boot 3.4
- Spring Security 6
- Spring Cloud Alibaba 2024.x
- MyBatis-Plus 3.5
- Redis 7.0
- Elasticsearch 8.x
- RabbitMQ 3.12
- MySQL 8.0
- Vue.js 3
- Docker / Kubernetes

## 三、核心功能实现

### 3.1 商品搜索（Elasticsearch）

```java
@Service
public class ProductSearchService {
    
    @Autowired
    private RestHighLevelClient esClient;
    
    /**
     * 商品搜索
     */
    public Result<Page<ProductES>> search(ProductSearchDTO dto) throws IOException {
        SearchRequest request = new SearchRequest("product");
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        
        // 构建查询
        BoolQueryBuilder boolQuery = QueryBuilders.boolQuery();
        
        // 关键词查询
        if (StringUtils.hasText(dto.getKeyword())) {
            boolQuery.must(QueryBuilders.multiMatchQuery(dto.getKeyword(), 
                "name", "description", "brand"));
        }
        
        // 分类过滤
        if (dto.getCategoryId() != null) {
            boolQuery.filter(QueryBuilders.termQuery("categoryId", dto.getCategoryId()));
        }
        
        // 价格区间
        if (dto.getMinPrice() != null || dto.getMaxPrice() != null) {
            RangeQueryBuilder rangeQuery = QueryBuilders.rangeQuery("price");
            if (dto.getMinPrice() != null) {
                rangeQuery.gte(dto.getMinPrice());
            }
            if (dto.getMaxPrice() != null) {
                rangeQuery.lte(dto.getMaxPrice());
            }
            boolQuery.filter(rangeQuery);
        }
        
        sourceBuilder.query(boolQuery);
        
        // 分页
        sourceBuilder.from((dto.getPage() - 1) * dto.getSize());
        sourceBuilder.size(dto.getSize());
        
        // 排序
        if ("sales".equals(dto.getSort())) {
            sourceBuilder.sort("sales", SortOrder.DESC);
        } else if ("price".equals(dto.getSort())) {
            sourceBuilder.sort("price", dto.getAsc() ? SortOrder.ASC : SortOrder.DESC);
        } else {
            sourceBuilder.sort("createTime", SortOrder.DESC);
        }
        
        // 高亮
        sourceBuilder.highlighter(new HighlightBuilder()
            .field("name")
            .field("description")
            .preTags("<em style='color:red'>")
            .postTags("</em>"));
        
        request.source(sourceBuilder);
        
        // 执行搜索
        SearchResponse response = esClient.search(request, RequestOptions.DEFAULT);
        
        // 解析结果
        List<ProductES> products = new ArrayList<>();
        for (SearchHit hit : response.getHits().getHits()) {
            ProductES product = JSON.parseObject(hit.getSourceAsString(), ProductES.class);
            
            // 处理高亮
            HighlightField nameField = hit.getHighlightFields().get("name");
            if (nameField != null) {
                product.setName(nameField.getFragments()[0].toString());
            }
            
            products.add(product);
        }
        
        // 构建分页结果
        Page<ProductES> page = new Page<>(dto.getPage(), dto.getSize(), 
            response.getHits().getTotalHits().value);
        page.setRecords(products);
        
        return Result.success(page);
    }
}
```

### 3.2 购物车（Redis）

```java
@Service
public class CartService {
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    private static final String CART_KEY_PREFIX = "cart:";
    
    /**
     * 添加商品到购物车
     */
    public Result<Void> addToCart(Long userId, CartDTO dto) {
        String key = CART_KEY_PREFIX + userId;
        
        // 检查商品是否存在
        Product product = productService.getById(dto.getProductId());
        if (product == null || product.getStock() < dto.getQuantity()) {
            return Result.error("商品库存不足");
        }
        
        // 使用 Hash 存储购物车
        HashOperations<String, String, CartItem> ops = redisTemplate.opsForHash();
        
        String skuKey = dto.getSkuId() != null ? dto.getSkuId().toString() : "default";
        
        if (ops.hasKey(key, skuKey)) {
            // 已存在，增加数量
            CartItem item = ops.get(key, skuKey);
            item.setQuantity(item.getQuantity() + dto.getQuantity());
            ops.put(key, skuKey, item);
        } else {
            // 新增
            CartItem item = new CartItem();
            item.setProductId(dto.getProductId());
            item.setSkuId(dto.getSkuId());
            item.setQuantity(dto.getQuantity());
            item.setPrice(product.getPrice());
            item.setName(product.getName());
            item.setImage(product.getImage());
            ops.put(key, skuKey, item);
        }
        
        return Result.success();
    }
    
    /**
     * 获取购物车
     */
    public Result<List<CartItem>> getCart(Long userId) {
        String key = CART_KEY_PREFIX + userId;
        HashOperations<String, String, CartItem> ops = redisTemplate.opsForHash();
        List<CartItem> items = ops.values(key);
        return Result.success(items);
    }
    
    /**
     * 更新购物车数量
     */
    public Result<Void> updateQuantity(Long userId, Long skuId, Integer quantity) {
        String key = CART_KEY_PREFIX + userId;
        HashOperations<String, String, CartItem> ops = redisTemplate.opsForHash();
        
        CartItem item = ops.get(key, skuId.toString());
        if (item != null) {
            item.setQuantity(quantity);
            ops.put(key, skuId.toString(), item);
        }
        
        return Result.success();
    }
    
    /**
     * 删除购物车商品
     */
    public Result<Void> deleteFromCart(Long userId, Long skuId) {
        String key = CART_KEY_PREFIX + userId;
        redisTemplate.opsForHash().delete(key, skuId.toString());
        return Result.success();
    }
    
    /**
     * 清空购物车
     */
    public Result<Void> clearCart(Long userId) {
        String key = CART_KEY_PREFIX + userId;
        redisTemplate.delete(key);
        return Result.success();
    }
}
```

### 3.3 创建订单

```java
@Service
public class OrderService {
    
    @Autowired
    private OrderMapper orderMapper;
    
    @Autowired
    private CartService cartService;
    
    @Autowired
    private ProductService productService;
    
    @Autowired
    private RabbitTemplate rabbitTemplate;
    
    /**
     * 创建订单
     */
    @Transactional
    public Result<Order> createOrder(Long userId, OrderDTO dto) {
        // 1. 获取购物车选中的商品
        List<CartItem> cartItems = cartService.getCheckedCart(userId);
        if (cartItems.isEmpty()) {
            return Result.error("购物车为空");
        }
        
        // 2. 检查库存
        for (CartItem item : cartItems) {
            Product product = productService.getById(item.getProductId());
            if (product.getStock() < item.getQuantity()) {
                return Result.error("商品 " + product.getName() + " 库存不足");
            }
        }
        
        // 3. 创建订单
        String orderNo = generateOrderNo();
        Order order = new Order();
        order.setOrderNo(orderNo);
        order.setUserId(userId);
        order.setTotalAmount(calculateTotal(cartItems));
        order.setPayAmount(order.getTotalAmount());
        order.setFreightAmount(new BigDecimal("0.00"));
        order.setStatus(OrderStatus.PENDING_PAYMENT);
        
        orderMapper.insert(order);
        
        // 4. 创建订单项
        List<OrderItem> orderItems = cartItems.stream().map(item -> {
            OrderItem orderItem = new OrderItem();
            orderItem.setOrderId(order.getId());
            orderItem.setProductId(item.getProductId());
            orderItem.setProductName(item.getName());
            orderItem.setProductPrice(item.getPrice());
            orderItem.setQuantity(item.getQuantity());
            orderItem.setSkuId(item.getSkuId());
            return orderItem;
        }).collect(Collectors.toList());
        
        orderItemService.saveBatch(orderItems);
        
        // 5. 扣减库存（发送消息）
        for (CartItem item : cartItems) {
            StockMessage message = new StockMessage();
            message.setProductId(item.getProductId());
            message.setQuantity(item.getQuantity());
            message.setOrderNo(orderNo);
            rabbitTemplate.convertAndSend("stock.exchange", "stock.decrease", message);
        }
        
        // 6. 清空购物车
        cartService.clearCart(userId);
        
        // 7. 发送延迟消息检查支付（30 分钟后）
        rabbitTemplate.convertAndSend("order.exchange", "order.timeout", orderNo,
            message -> {
                message.getMessageProperties().setDelay(30 * 60 * 1000);
                return message;
            });
        
        return Result.success(order);
    }
    
    /**
     * 生成订单号
     */
    private String generateOrderNo() {
        return "ORD" + System.currentTimeMillis() + 
               ThreadLocalRandom.current().nextInt(1000, 9999);
    }
    
    /**
     * 计算总金额
     */
    private BigDecimal calculateTotal(List<CartItem> items) {
        return items.stream()
            .map(item -> item.getPrice().multiply(new BigDecimal(item.getQuantity())))
            .reduce(BigDecimal.ZERO, BigDecimal::add);
    }
}
```

### 3.4 订单超时取消

```java
@Component
public class OrderTimeoutListener {
    
    @Autowired
    private OrderService orderService;
    
    @RabbitListener(queues = "order.timeout.queue")
    public void handleTimeout(String orderNo) {
        Order order = orderService.getByOrderNo(orderNo);
        
        if (order != null && order.getStatus() == OrderStatus.PENDING_PAYMENT) {
            // 取消订单
            orderService.cancelOrder(order.getId());
            
            // 恢复库存
            List<OrderItem> items = orderItemService.getByOrderId(order.getId());
            for (OrderItem item : items) {
                productService.increaseStock(item.getProductId(), item.getQuantity());
            }
            
            System.out.println("订单超时取消：" + orderNo);
        }
    }
}
```

### 3.5 支付回调

```java
@RestController
@RequestMapping("/api/payment")
public class PaymentController {
    
    @Autowired
    private OrderService orderService;
    
    /**
     * 支付宝回调
     */
    @PostMapping("/alipay/callback")
    public String alipayCallback(@RequestParam Map<String, String> params) {
        String orderNo = params.get("out_trade_no");
        String tradeNo = params.get("trade_no");
        String tradeStatus = params.get("trade_status");
        
        if ("TRADE_SUCCESS".equals(tradeStatus)) {
            // 更新订单状态
            orderService.paySuccess(orderNo, tradeNo);
            return "success";
        }
        
        return "fail";
    }
    
    /**
     * 微信支付回调
     */
    @PostMapping("/wechat/callback")
    public Map<String, String> wechatCallback(@RequestBody String xmlData) {
        // 解析 XML
        Map<String, String> params = parseXml(xmlData);
        
        String orderNo = params.get("out_trade_no");
        String transactionId = params.get("transaction_id");
        String returnCode = params.get("return_code");
        
        Map<String, String> result = new HashMap<>();
        
        if ("SUCCESS".equals(returnCode)) {
            orderService.paySuccess(orderNo, transactionId);
            result.put("return_code", "SUCCESS");
            result.put("return_msg", "OK");
        } else {
            result.put("return_code", "FAIL");
        }
        
        return result;
    }
}
```

## 四、库存扣减（分布式锁）

```java
@Service
public class StockService {
    
    @Autowired
    private RedissonClient redissonClient;
    
    @Autowired
    private ProductMapper productMapper;
    
    /**
     * 扣减库存（分布式锁）
     */
    public Result<Void> decreaseStock(Long productId, Integer quantity) {
        String lockKey = "stock:lock:" + productId;
        RLock lock = redissonClient.getLock(lockKey);
        
        if (lock.tryLock()) {
            try {
                Product product = productMapper.selectById(productId);
                if (product.getStock() < quantity) {
                    return Result.error("库存不足");
                }
                
                productMapper.decreaseStock(productId, quantity);
                return Result.success();
            } finally {
                lock.unlock();
            }
        } else {
            return Result.error("系统繁忙");
        }
    }
}
```
