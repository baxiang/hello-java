# 14.1 博客系统项目

## 一、项目介绍

### 1.1 项目概述

博客系统是一个内容发布平台，用户可以发布文章、管理分类、添加标签、评论互动。

**技术栈**：
- Spring Boot 2.7
- MyBatis-Plus
- MySQL 8.0
- Redis 6.0
- Thymeleaf / Vue.js

### 1.2 功能模块

```
├── 用户模块
│   ├── 注册登录
│   ├── 个人信息
```

## 二、数据库设计

### 2.1 用户表

```sql
CREATE TABLE `user` (
    `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
    `username` VARCHAR(50) NOT NULL UNIQUE,
    `password` VARCHAR(100) NOT NULL,
    `email` VARCHAR(100),
    `avatar` VARCHAR(255),
    `create_time` DATETIME DEFAULT CURRENT_TIMESTAMP,
    `update_time` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);
```

### 2.2 文章表

```sql
CREATE TABLE `article` (
    `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
    `title` VARCHAR(200) NOT NULL,
    `summary` VARCHAR(500),
    `content` TEXT NOT NULL,
    `author_id` BIGINT NOT NULL,
    `category_id` BIGINT,
    `view_count` INT DEFAULT 0,
    `comment_count` INT DEFAULT 0,
    `status` TINYINT DEFAULT 1,  -- 1:发布 0:草稿
    `create_time` DATETIME DEFAULT CURRENT_TIMESTAMP,
    `update_time` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (`author_id`) REFERENCES `user`(`id`),
    FOREIGN KEY (`category_id`) REFERENCES `category`(`id`)
);
```

### 2.3 分类表

```sql
CREATE TABLE `category` (
    `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
    `name` VARCHAR(50) NOT NULL,
    `description` VARCHAR(200),
    `create_time` DATETIME DEFAULT CURRENT_TIMESTAMP
);
```

### 2.4 标签表

```sql
CREATE TABLE `tag` (
    `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
    `name` VARCHAR(50) NOT NULL UNIQUE,
    `create_time` DATETIME DEFAULT CURRENT_TIMESTAMP
);
```

### 2.5 文章标签关联表

```sql
CREATE TABLE `article_tag` (
    `article_id` BIGINT NOT NULL,
    `tag_id` BIGINT NOT NULL,
    PRIMARY KEY (`article_id`, `tag_id`),
    FOREIGN KEY (`article_id`) REFERENCES `article`(`id`),
    FOREIGN KEY (`tag_id`) REFERENCES `tag`(`id`)
);
```

### 2.6 评论表

```sql
CREATE TABLE `comment` (
    `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
    `article_id` BIGINT NOT NULL,
    `user_id` BIGINT NOT NULL,
    `parent_id` BIGINT DEFAULT 0,
    `content` TEXT NOT NULL,
    `create_time` DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (`article_id`) REFERENCES `article`(`id`),
    FOREIGN KEY (`user_id`) REFERENCES `user`(`id`)
);
```

## 四、核心功能实现

### 4.1 用户注册

```java
@Service
public class UserServiceImpl extends ServiceImpl<UserMapper, User> implements UserService {
    
    @Autowired
    private PasswordEncoder passwordEncoder;
    
    @Override
    @Transactional
    public Result<User> register(RegisterDTO dto) {
        // 检查用户名是否存在
        User existUser = lambdaQuery()
            .eq(User::getUsername, dto.getUsername())
            .one();
        
        if (existUser != null) {
            return Result.error("用户名已存在");
        }
        
        // 创建用户
        User user = new User();
        user.setUsername(dto.getUsername());
        user.setPassword(passwordEncoder.encode(dto.getPassword()));
        user.setEmail(dto.getEmail());
        
        baseMapper.insert(user);
        
        return Result.success(user);
    }
}
```

### 4.2 用户登录

```java
@Service
public class UserServiceImpl extends ServiceImpl<UserMapper, User> implements UserService {
    
    @Autowired
    private JwtUtil jwtUtil;
    
    @Override
    public Result<LoginVO> login(LoginDTO dto) {
        // 查询用户
        User user = lambdaQuery()
            .eq(User::getUsername, dto.getUsername())
            .one();
        
        if (user == null) {
            return Result.error("用户不存在");
        }
        
        // 验证密码
        if (!passwordEncoder.matches(dto.getPassword(), user.getPassword())) {
            return Result.error("密码错误");
        }
        
        // 生成 Token
        String token = jwtUtil.generateToken(user.getId(), user.getUsername());
        
        LoginVO vo = new LoginVO();
        vo.setToken(token);
        vo.setUser(user);
        
        return Result.success(vo);
    }
}
```

### 4.3 文章发布

```java
@Service
public class ArticleServiceImpl extends ServiceImpl<ArticleMapper, Article> implements ArticleService {
    
    @Override
    @Transactional
    public Result<Article> publish(ArticleDTO dto, Long userId) {
        Article article = new Article();
        article.setTitle(dto.getTitle());
        article.setSummary(dto.getSummary());
        article.setContent(dto.getContent());
        article.setAuthorId(userId);
        article.setCategoryId(dto.getCategoryId());
        article.setStatus(dto.getStatus());
        
        baseMapper.insert(article);
        
        // 保存标签关联
        if (dto.getTagIds() != null && !dto.getTagIds().isEmpty()) {
            List<ArticleTag> tags = dto.getTagIds().stream()
                .map(tagId -> {
                    ArticleTag at = new ArticleTag();
                    at.setArticleId(article.getId());
                    at.setTagId(tagId);
                    return at;
                })
                .collect(Collectors.toList());
            articleTagService.saveBatch(tags);
        }
        
        return Result.success(article);
    }
}
```

### 4.4 文章列表

```java
@Service
public class ArticleServiceImpl extends ServiceImpl<ArticleMapper, Article> implements ArticleService {
    
    @Override
    public Result<IPage<ArticleVO>> getArticles(int page, int size, Long categoryId, String keyword) {
        Page<Article> articlePage = new Page<>(page, size);
        
        LambdaQueryWrapper<Article> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(Article::getStatus, 1);  // 只查询已发布
        
        if (categoryId != null) {
            wrapper.eq(Article::getCategoryId, categoryId);
        }
        
        if (StringUtils.hasText(keyword)) {
            wrapper.and(w -> w
                .like(Article::getTitle, keyword)
                .or()
                .like(Article::getSummary, keyword)
            );
        }
        
        wrapper.orderByDesc(Article::getCreateTime);
        
        IPage<Article> result = baseMapper.selectPage(articlePage, wrapper);
        
        // 转换为 VO
        IPage<ArticleVO> voPage = result.convert(article -> {
            ArticleVO vo = new ArticleVO();
            BeanUtils.copyProperties(article, vo);
            
            // 查询作者信息
            User author = userService.getById(article.getAuthorId());
            vo.setAuthorName(author.getUsername());
            vo.setAuthorAvatar(author.getAvatar());
            
            // 查询分类信息
            Category category = categoryService.getById(article.getCategoryId());
            vo.setCategoryName(category != null ? category.getName() : null);
            
            // 查询标签
            List<Tag> tags = articleTagService.getTagsByArticleId(article.getId());
            vo.setTags(tags);
            
            return vo;
        });
        
        return Result.success(voPage);
    }
}
```

### 4.5 发表评论

```java
@Service
public class CommentServiceImpl extends ServiceImpl<CommentMapper, Comment> implements CommentService {
    
    @Override
    @Transactional
    public Result<Comment> publish(CommentDTO dto, Long userId) {
        Comment comment = new Comment();
        comment.setArticleId(dto.getArticleId());
        comment.setUserId(userId);
        comment.setParentId(dto.getParentId() != null ? dto.getParentId() : 0);
        comment.setContent(dto.getContent());
        
        baseMapper.insert(comment);
        
        // 更新文章评论数
        articleService.lambdaUpdate()
            .setSql("comment_count = comment_count + 1")
            .eq(Article::getId, dto.getArticleId())
            .update();
        
        return Result.success(comment);
    }
    
    @Override
    public Result<List<CommentVO>> getComments(Long articleId) {
        List<Comment> comments = lambdaQuery()
            .eq(Comment::getArticleId, articleId)
            .orderByAsc(Comment::getCreateTime)
            .list();
        
        List<CommentVO> voList = comments.stream().map(comment -> {
            CommentVO vo = new CommentVO();
            BeanUtils.copyProperties(comment, vo);
            
            User user = userService.getById(comment.getUserId());
            vo.setUsername(user.getUsername());
            vo.setAvatar(user.getAvatar());
            
            // 查询回复
            if (comment.getParentId() == 0) {
                List<Comment> replies = lambdaQuery()
                    .eq(Comment::getParentId, comment.getId())
                    .list();
                vo.setReplies(replies.stream().map(r -> {
                    CommentVO rVo = new CommentVO();
                    BeanUtils.copyProperties(r, rVo);
                    User rUser = userService.getById(r.getUserId());
                    rVo.setUsername(rUser.getUsername());
                    return rVo;
                }).collect(Collectors.toList()));
            }
            
            return vo;
        }).collect(Collectors.toList());
        
        return Result.success(voList);
    }
}
```

## 五、缓存优化

### 5.1 Redis 缓存文章

```java
@Service
public class ArticleServiceImpl extends ServiceImpl<ArticleMapper, Article> implements ArticleService {
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    @Override
    public Result<ArticleVO> getArticleById(Long id) {
        String key = "article:" + id;
        
        // 查缓存
        ArticleVO vo = (ArticleVO) redisTemplate.opsForValue().get(key);
        if (vo != null) {
            return Result.success(vo);
        }
        
        // 查数据库
        Article article = baseMapper.selectById(id);
        if (article == null) {
            return Result.error("文章不存在");
        }
        
        // 增加阅读量
        baseMapper.incrementViewCount(id);
        
        // 转换为 VO
        vo = convertToVO(article);
        
        // 写入缓存，30 分钟过期
        redisTemplate.opsForValue().set(key, vo, 30, TimeUnit.MINUTES);
        
        return Result.success(vo);
    }
}
```

### 5.2 缓存热点文章

```java
@Service
public class ArticleService {
    
    @Scheduled(cron = "0 */10 * * * *")  // 每 10 分钟执行
    public void cacheHotArticles() {
        // 查询热门文章（阅读量前 10）
        List<Article> hotArticles = baseMapper.selectHotArticles(10);
        
        for (Article article : hotArticles) {
            String key = "article:hot:" + article.getId();
            redisTemplate.opsForValue().set(key, article, 1, TimeUnit.HOURS);
        }
    }
}
```

## 六、安全配置

### 6.1 JWT 拦截器

```java
@Component
public class JwtInterceptor implements HandlerInterceptor {
    
    @Autowired
    private JwtUtil jwtUtil;
    
    @Override
    public boolean preHandle(HttpServletRequest request,
                           HttpServletResponse response,
                           Object handler) throws Exception {
        
        String token = request.getHeader("Authorization");
        
        if (token == null || !token.startsWith("Bearer ")) {
            response.setStatus(401);
            return false;
        }
        
        try {
            Claims claims = jwtUtil.parseToken(token.substring(7));
            Long userId = claims.get("userId", Long.class);
            request.setAttribute("userId", userId);
            return true;
        } catch (Exception e) {
            response.setStatus(401);
            return false;
        }
    }
}
```
