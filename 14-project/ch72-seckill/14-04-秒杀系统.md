# 14.4 秒杀系统

## 一、项目介绍

### 1.1 项目概述

秒杀系统是一个高并发场景下的商品抢购平台，需要处理瞬时大流量、防止超卖、保证系统稳定。

**技术栈**：
- Spring Boot 3.4
- Redis 7.0（缓存 + 分布式锁）
- RabbitMQ 3.12（异步下单）
- Sentinel 1.8（限流熔断）
- MySQL 8.0
- GraalVM 原生编译（可选）

### 1.2 核心挑战

| 挑战 | 解决方案 |
|------|----------|
| 高并发读 | Redis 缓存 |
| 高并发写 | 消息队列异步 |
| 超卖问题 | 分布式锁 + 乐观锁 |
| 恶意请求 | 限流 + 验证码 |
| 系统雪崩 | 熔断降级 |

### 1.3 系统架构

```
用户请求
   ↓
Nginx 限流
   ↓
Gateway 网关
   ↓
Sentinel 限流
   ↓
Redis 预减库存
   ↓
RabbitMQ 异步下单
   ↓
MySQL 最终扣减
```

## 三、核心功能实现

### 3.1 秒杀准备（预热缓存）

```java
@Service
public class SeckillService {
    
    @Autowired
    private SeckillProductMapper seckillProductMapper;
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    /**
     * 活动开始前预热缓存
     */
    @PostConstruct
    public void warmUpCache() {
        List<SeckillProduct> products = seckillProductMapper.selectList(
            new LambdaQueryWrapper<SeckillProduct>()
                .eq(SeckillProduct::getStatus, 1)
        );
        
        for (SeckillProduct product : products) {
            // 缓存商品信息
            String productKey = "seckill:product:" + product.getId();
            redisTemplate.opsForValue().set(productKey, product);
            
            // 缓存库存
            String stockKey = "seckill:stock:" + product.getId();
            redisTemplate.opsForValue().set(stockKey, product.getStock());
            
            // 缓存活动时间
            String timeKey = "seckill:time:" + product.getId();
            Map<String, Object> timeMap = new HashMap<>();
            timeMap.put("start", product.getStartTime().toInstant(ZoneOffset.of("+8")).toEpochMilli());
            timeMap.put("end", product.getEndTime().toInstant(ZoneOffset.of("+8")).toEpochMilli());
            redisTemplate.opsForValue().set(timeKey, timeMap);
        }
    }
}
```

### 3.2 秒杀接口

```java
@Service
public class SeckillService {
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    @Autowired
    private RabbitTemplate rabbitTemplate;
    
    @Autowired
    private IdWorker idWorker;
    
    /**
     * 秒杀下单
     */
    @SentinelResource(
        value = "seckill",
        blockHandler = "handleBlock"
    )
    public Result<String> seckill(Long userId, Long productId) {
        // 1. 验证活动时间
        if (!isActivityTime(productId)) {
            return Result.error("活动未开始或已结束");
        }
        
        // 2. 验证是否重复购买
        String orderKey = "seckill:order:" + productId + ":" + userId;
        if (Boolean.TRUE.equals(redisTemplate.hasKey(orderKey))) {
            return Result.error("您已购买过该商品");
        }
        
        // 3. 预减库存（Redis 原子操作）
        String stockKey = "seckill:stock:" + productId;
        Long stock = redisTemplate.opsForValue().decrement(stockKey);
        
        if (stock < 0) {
            // 库存不足，恢复
            redisTemplate.opsForValue().increment(stockKey);
            return Result.error("库存不足");
        }
        
        // 4. 生成订单号
        String orderNo = "SK" + idWorker.nextId();
        
        // 5. 发送消息到 MQ
        SeckillMessage message = new SeckillMessage();
        message.setUserId(userId);
        message.setProductId(productId);
        message.setOrderNo(orderNo);
        
        rabbitTemplate.convertAndSend("seckill.exchange", "seckill.order", message);
        
        // 6. 标记已购买
        redisTemplate.opsForValue().set(orderKey, orderNo, 30, TimeUnit.MINUTES);
        
        return Result.success(orderNo);
    }
    
    /**
     * 限流处理
     */
    public Result<String> handleBlock(Long userId, Long productId, BlockException ex) {
        return Result.error("系统繁忙，请稍后重试");
    }
    
    /**
     * 验证活动时间
     */
    private boolean isActivityTime(Long productId) {
        String timeKey = "seckill:time:" + productId;
        Map<String, Object> timeMap = (Map<String, Object>) redisTemplate.opsForValue().get(timeKey);
        
        if (timeMap == null) {
            return false;
        }
        
        long now = System.currentTimeMillis();
        long start = (Long) timeMap.get("start");
        long end = (Long) timeMap.get("end");
        
        return now >= start && now <= end;
    }
}
```

### 3.3 异步下单消费者

```java
@Component
public class SeckillOrderConsumer {
    
    @Autowired
    private SeckillOrderMapper seckillOrderMapper;
    
    @Autowired
    private SeckillProductMapper seckillProductMapper;
    
    @Autowired
    private SeckillLogMapper seckillLogMapper;
    
    @RabbitListener(queues = "seckill.order.queue")
    public void handleOrder(SeckillMessage message) {
        Long userId = message.getUserId();
        Long productId = message.getProductId();
        String orderNo = message.getOrderNo();
        
        try {
            // 1. 检查是否重复下单
            SeckillOrder existOrder = seckillOrderMapper.selectOne(
                new LambdaQueryWrapper<SeckillOrder>()
                    .eq(SeckillOrder::getUserId, userId)
                    .eq(SeckillOrder::getProductId, productId)
            );
            
            if (existOrder != null) {
                return;
            }
            
            // 2. 获取商品信息
            SeckillProduct product = seckillProductMapper.selectById(productId);
            if (product == null || product.getStock() <= 0) {
                throw new BusinessException("商品不存在或库存不足");
            }
            
            // 3. 乐观锁扣减库存
            int updated = seckillProductMapper.update(null,
                new LambdaUpdateWrapper<SeckillProduct>()
                    .eq(SeckillProduct::getId, productId)
                    .gt(SeckillProduct::getStock, 0)
                    .setSql("stock = stock - 1")
                    .setSql("version = version + 1")
            );
            
            if (updated == 0) {
                throw new BusinessException("库存不足");
            }
            
            // 4. 创建订单
            SeckillOrder order = new SeckillOrder();
            order.setUserId(userId);
            order.setProductId(productId);
            order.setOrderNo(orderNo);
            order.setSeckillPrice(product.getSeckillPrice());
            order.setStatus(OrderStatus.PENDING_PAYMENT);
            
            seckillOrderMapper.insert(order);
            
            // 5. 记录成功日志
            SeckillLog log = new SeckillLog();
            log.setUserId(userId);
            log.setProductId(productId);
            log.setStatus(1);
            log.setMessage("秒杀成功");
            seckillLogMapper.insert(log);
            
        } catch (Exception e) {
            // 记录失败日志
            SeckillLog log = new SeckillLog();
            log.setUserId(userId);
            log.setProductId(productId);
            log.setStatus(0);
            log.setMessage(e.getMessage());
            seckillLogMapper.insert(log);
            
            // 恢复库存（异步）
            // ...
        }
    }
}
```

### 3.4 查询秒杀结果

```java
@Service
public class SeckillService {
    
    @Autowired
    private SeckillOrderMapper seckillOrderMapper;
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    /**
     * 查询秒杀结果
     */
    public Result<SeckillResult> getResult(Long userId, Long productId) {
        // 1. 查询订单
        SeckillOrder order = seckillOrderMapper.selectOne(
            new LambdaQueryWrapper<SeckillOrder>()
                .eq(SeckillOrder::getUserId, userId)
                .eq(SeckillOrder::getProductId, productId)
        );
        
        if (order != null) {
            SeckillResult result = new SeckillResult();
            result.setSuccess(true);
            result.setOrderNo(order.getOrderNo());
            result.setStatus(order.getStatus());
            return Result.success(result);
        }
        
        // 2. 检查是否还在排队
        String stockKey = "seckill:stock:" + productId;
        Long stock = (Long) redisTemplate.opsForValue().get(stockKey);
        
        // 如果库存还有剩余，说明可能还在处理中
        // 这里简化处理，直接返回失败
        SeckillResult result = new SeckillResult();
        result.setSuccess(false);
        result.setMessage("秒杀失败");
        
        return Result.success(result);
    }
}
```

## 四、防刷机制

### 4.1 图形验证码

```java
@RestController
@RequestMapping("/api/seckill")
public class SeckillController {
    
    @Autowired
    private CaptchaService captchaService;
    
    /**
     * 获取验证码
     */
    @GetMapping("/captcha")
    public void getCaptcha(HttpServletResponse response, 
                          @RequestParam Long productId) throws IOException {
        String uuid = UUID.randomUUID().toString();
        
        // 生成验证码
        Captcha captcha = captchaService.generateCaptcha();
        
        // 缓存验证码
        redisTemplate.opsForValue().set(
            "seckill:captcha:" + productId + ":" + uuid,
            captcha.getCode(),
            5, TimeUnit.MINUTES
        );
        
        // 返回验证码图片
        response.setContentType("image/jpeg");
        captcha.getImage().write(response.getOutputStream());
    }
    
    /**
     * 验证验证码
     */
    @PostMapping("/verify")
    public Result<Boolean> verifyCaptcha(@RequestParam Long productId,
                                         @RequestParam String uuid,
                                         @RequestParam String code) {
        String key = "seckill:captcha:" + productId + ":" + uuid;
        String cachedCode = (String) redisTemplate.opsForValue().get(key);
        
        if (cachedCode == null) {
            return Result.error("验证码已过期");
        }
        
        if (!cachedCode.equalsIgnoreCase(code)) {
            return Result.success(false);
        }
        
        // 验证成功后删除验证码
        redisTemplate.delete(key);
        
        return Result.success(true);
    }
}
```

## 五、性能优化

### 5.1 库存预热

```java
@Component
public class SeckillPreheat implements ApplicationListener<ApplicationReadyEvent> {
    
    @Autowired
    private SeckillProductMapper seckillProductMapper;
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    @Override
    public void onApplicationEvent(ApplicationReadyEvent event) {
        // 活动开始前预热库存
        List<SeckillProduct> products = seckillProductMapper.selectList(
            new LambdaQueryWrapper<SeckillProduct>()
                .ge(SeckillProduct::getStartTime, LocalDateTime.now())
                .le(SeckillProduct::getStartTime, LocalDateTime.now().plusHours(1))
        );
        
        for (SeckillProduct product : products) {
            String stockKey = "seckill:stock:" + product.getId();
            redisTemplate.opsForValue().set(stockKey, product.getStock());
        }
    }
}
```
