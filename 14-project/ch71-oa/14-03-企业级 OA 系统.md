# 14.3 企业级 OA 系统

## 一、项目介绍

### 1.1 项目概述

企业级 OA（Office Automation）系统是一个办公自动化平台，包含员工管理、考勤、请假、审批等功能。

**技术栈**：
- Spring Boot 3.4
- Spring Security 6 + JWT
- MyBatis-Plus 3.5
- Vue.js 3 + Element Plus
- MySQL 8.0
- Redis 7.0
- Docker / Kubernetes

### 1.2 功能模块

```
OA 系统
├── 系统管理
│   ├── 用户管理
│   ├── 角色管理
│   ├── 权限管理

├── 考勤管理

```

### 3.1 Spring Security 配置

```java
@Configuration
@EnableWebSecurity
@EnableGlobalMethodSecurity(prePostEnabled = true)
public class SecurityConfig {
    
    @Autowired
    private UserDetailsService userDetailsService;
    
    @Autowired
    private JwtAuthenticationEntryPoint jwtEntryPoint;
    
    @Bean
    public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
        http
            .csrf().disable()
            .exceptionHandling()
                .authenticationEntryPoint(jwtEntryPoint)
            .and()
            .sessionManagement()
                .sessionCreationPolicy(SessionCreationPolicy.STATELESS)
            .and()
            .authorizeRequests()
                .antMatchers("/api/auth/**").permitAll()
                .antMatchers("/api/public/**").permitAll()
                .antMatchers("/api/admin/**").hasRole("ADMIN")
                .anyRequest().authenticated();
        
        http.addFilterBefore(jwtAuthenticationFilter(), 
            UsernamePasswordAuthenticationFilter.class);
        
        return http.build();
    }
    
    @Bean
    public PasswordEncoder passwordEncoder() {
        return new BCryptPasswordEncoder();
    }
    
    @Bean
    public JwtAuthenticationFilter jwtAuthenticationFilter() {
        return new JwtAuthenticationFilter();
    }
}
```

### 3.2 用户详情服务

```java
@Service
public class UserDetailsServiceImpl implements UserDetailsService {
    
    @Autowired
    private UserMapper userMapper;
    
    @Autowired
    private RoleMapper roleMapper;
    
    @Autowired
    private PermissionMapper permissionMapper;
    
    @Override
    public UserDetails loadUserByUsername(String username) 
            throws UsernameNotFoundException {
        // 查询用户
        User user = userMapper.selectByUsername(username);
        if (user == null) {
            throw new UsernameNotFoundException("用户不存在");
        }
        
        // 查询角色
        List<Role> roles = roleMapper.selectByUserId(user.getId());
        List<GrantedAuthority> authorities = roles.stream()
            .map(role -> new SimpleGrantedAuthority("ROLE_" + role.getCode()))
            .collect(Collectors.toList());
        
        // 查询权限
        List<Permission> permissions = permissionMapper.selectByUserId(user.getId());
        for (Permission permission : permissions) {
            if (StringUtils.hasText(permission.getPermission())) {
                authorities.add(new SimpleGrantedAuthority(permission.getPermission()));
            }
        }
        
        return new org.springframework.security.core.userdetails.User(
            user.getUsername(),
            user.getPassword(),
            user.getStatus() == 1,
            true, true, true,
            authorities
        );
    }
}
```

### 3.3 动态菜单权限

```java
@Service
public class PermissionService {
    
    @Autowired
    private PermissionMapper permissionMapper;
    
    /**
     * 获取用户菜单
     */
    public List<MenuVO> getUserMenus(Long userId) {
        // 查询用户权限
        List<Permission> permissions = permissionMapper.selectByUserId(userId);
        
        // 转换为菜单树
        return buildMenuTree(permissions, 0L);
    }
    
    /**
     * 构建菜单树
     */
    private List<MenuVO> buildMenuTree(List<Permission> permissions, Long parentId) {
        return permissions.stream()
            .filter(p -> p.getType() == 1 && p.getParentId().equals(parentId))
            .map(permission -> {
                MenuVO menu = new MenuVO();
                menu.setId(permission.getId());
                menu.setName(permission.getName());
                menu.setPath(permission.getPath());
                menu.setComponent(permission.getComponent());
                menu.setIcon(permission.getIcon());
                
                // 递归查询子菜单
                menu.setChildren(buildMenuTree(permissions, permission.getId()));
                
                return menu;
            })
            .sorted(Comparator.comparingInt(MenuVO::getSort))
            .collect(Collectors.toList());
    }
}
```

### 3.4 打卡签到

```java
@Service
public class AttendanceService {
    
    @Autowired
    private AttendanceMapper attendanceMapper;
    
    /**
     * 打卡签到
     */
    @Transactional
    public Result<Attendance> clockIn(Long userId, LocalDateTime clockInTime) {
        LocalDate today = LocalDate.now();
        
        // 检查是否已打卡
        Attendance attendance = attendanceMapper.selectByUserIdAndDate(userId, today);
        
        if (attendance != null) {
            return Result.error("今日已打卡");
        }
        
        // 创建打卡记录
        attendance = new Attendance();
        attendance.setUserId(userId);
        attendance.setDate(today);
        attendance.setClockIn(clockInTime);
        
        // 判断是否迟到（9:30 后）
        LocalTime clockInTimeOnly = clockInTime.toLocalTime();
        if (clockInTimeOnly.isAfter(LocalTime.of(9, 30))) {
            attendance.setStatus(AttendanceStatus.LATE);
        } else {
            attendance.setStatus(AttendanceStatus.NORMAL);
        }
        
        attendanceMapper.insert(attendance);
        
        return Result.success(attendance);
    }
    
    /**
     * 打卡签退
     */
    @Transactional
    public Result<Attendance> clockOut(Long userId, LocalDateTime clockOutTime) {
        LocalDate today = LocalDate.now();
        Attendance attendance = attendanceMapper.selectByUserIdAndDate(userId, today);
        
        if (attendance == null) {
            return Result.error("请先签到");
        }
        
        if (attendance.getClockOut() != null) {
            return Result.error("今日已签退");
        }
        
        attendance.setClockOut(clockOutTime);
        
        // 判断是否早退（18:00 前）
        LocalTime clockOutTimeOnly = clockOutTime.toLocalTime();
        if (clockOutTimeOnly.isBefore(LocalTime.of(18, 0))) {
            attendance.setStatus(AttendanceStatus.EARLY_LEAVE);
        }
        
        attendanceMapper.updateById(attendance);
        
        return Result.success(attendance);
    }
}
```

### 3.5 请假申请

```java
@Service
public class LeaveService {
    
    @Autowired
    private LeaveMapper leaveMapper;
    
    @Autowired
    private UserMapper userMapper;
    
    /**
     * 提交请假申请
     */
    @Transactional
    public Result<LeaveRequest> applyLeave(Long userId, LeaveDTO dto) {
        // 计算请假天数
        BigDecimal days = calculateDays(dto.getStartTime(), dto.getEndTime());
        
        LeaveRequest request = new LeaveRequest();
        request.setUserId(userId);
        request.setType(dto.getType());
        request.setStartTime(dto.getStartTime());
        request.setEndTime(dto.getEndTime());
        request.setDays(days);
        request.setReason(dto.getReason());
        request.setStatus(LeaveStatus.PENDING);
        
        leaveMapper.insert(request);
        
        // 通知审批人
        User user = userMapper.selectById(userId);
        notifyApprover(user.getManagerId(), user.getRealName(), dto.getType(), days);
        
        return Result.success(request);
    }
    
    /**
     * 审批请假
     */
    @Transactional
    public Result<Void> approveLeave(Long approverId, Long requestId, boolean approved, 
                                     String remark) {
        LeaveRequest request = leaveMapper.selectById(requestId);
        
        if (request == null) {
            return Result.error("请假申请不存在");
        }
        
        if (request.getStatus() != LeaveStatus.PENDING) {
            return Result.error("申请已处理");
        }
        
        request.setStatus(approved ? LeaveStatus.APPROVED : LeaveStatus.REJECTED);
        request.setApproverId(approverId);
        request.setApproveTime(LocalDateTime.now());
        
        leaveMapper.updateById(request);
        
        // 通知申请人
        notifyApplicant(request.getUserId(), approved, remark);
        
        return Result.success();
    }
    
    /**
     * 计算请假天数
     */
    private BigDecimal calculateDays(LocalDateTime start, LocalDateTime end) {
        long milliseconds = end.toInstant().toEpochMilli() - start.toInstant().toEpochMilli();
        return new BigDecimal(milliseconds).divide(new BigDecimal(86400000), 2, RoundingMode.HALF_UP);
    }
}
```

### 3.6 考勤统计

```java
@Service
public class AttendanceService {
    
    /**
     * 月度考勤统计
     */
    public Result<AttendanceStats> getMonthlyStats(Long userId, YearMonth yearMonth) {
        LocalDate startDate = yearMonth.atDay(1);
        LocalDate endDate = yearMonth.atEndOfMonth();
        
        List<Attendance> attendances = attendanceMapper.selectByUserIdAndDateRange(
            userId, startDate, endDate);
        
        AttendanceStats stats = new AttendanceStats();
        stats.setUserId(userId);
        stats.setYearMonth(yearMonth);
        
        int normalDays = 0;
        int lateDays = 0;
        int earlyLeaveDays = 0;
        int absentDays = 0;
        
        for (Attendance attendance : attendances) {
            switch (attendance.getStatus()) {
                case NORMAL: normalDays++; break;
                case LATE: lateDays++; break;
                case EARLY_LEAVE: earlyLeaveDays++; break;
                case ABSENT: absentDays++; break;
            }
        }
        
        stats.setNormalDays(normalDays);
        stats.setLateDays(lateDays);
        stats.setEarlyLeaveDays(earlyLeaveDays);
        stats.setAbsentDays(absentDays);
        stats.setTotalDays(attendances.size());
        
        return Result.success(stats);
    }
    
    /**
     * 部门考勤统计
     */
    public Result<List<DeptAttendanceStats>> getDeptStats(Long deptId, YearMonth yearMonth) {
        return Result.success(attendanceMapper.selectDeptStats(deptId, yearMonth));
    }
}
```

### 4.2 获取公告列表

```java
@Service
public class NoticeService {
    
    public Result<IPage<NoticeVO>> getNotices(int page, int size, Integer type) {
        Page<Notice> noticePage = new Page<>(page, size);
        
        LambdaQueryWrapper<Notice> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(Notice::getStatus, 1);  // 只查询已发布
        
        if (type != null) {
            wrapper.eq(Notice::getType, type);
        }
        
        wrapper.orderByDesc(Notice::getPublishTime);
        
        IPage<Notice> result = noticeMapper.selectPage(noticePage, wrapper);
        
        IPage<NoticeVO> voPage = result.convert(notice -> {
            NoticeVO vo = new NoticeVO();
            BeanUtils.copyProperties(notice, vo);
            
            User publisher = userMapper.selectById(notice.getPublisherId());
            vo.setPublisherName(publisher.getRealName());
            
            return vo;
        });
        
        return Result.success(voPage);
    }
}
```
