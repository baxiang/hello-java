# 79.8 应用上下文（ApplicationContext）

## 一、ApplicationContext 接口

扩展 BeanFactory，提供更完整的功能。

```java
package com.example.minispring.context;

import com.example.minispring.beans.factory.BeanFactory;

/**
 * 应用上下文接口：扩展 BeanFactory
 */
public interface ApplicationContext extends BeanFactory {
    
    /**
     * 获取 Bean 名称数组
     */
    String[] getBeanDefinitionNames();
    
    /**
     * 关闭上下文
     */
    void close();
    
    /**
     * 添加钩子，在 JVM 关闭时关闭上下文
     */
    void registerShutdownHook();
}
```

## 二、AbstractApplicationContext 抽象类

```java
package com.example.minispring.context.support;

import com.example.minispring.beans.factory.support.DefaultListableBeanFactory;
import com.example.minispring.context.ApplicationContext;

/**
 * 应用上下文抽象类
 */
public abstract class AbstractApplicationContext 
        extends DefaultListableBeanFactory 
        implements ApplicationContext {
    
    /** 是否已刷新 */
    private boolean refreshed = false;
    
    /**
     * 刷新容器
     */
    public void refresh() {
        if (refreshed) {
            throw new IllegalStateException("ApplicationContext can only be refreshed once");
        }
        refreshed = true;
        
        // 1. 准备刷新
        prepareRefresh();
        
        // 2. 获取 BeanFactory
        DefaultListableBeanFactory beanFactory = obtainFreshBeanFactory();
        
        // 3. 准备 BeanFactory
        prepareBeanFactory(beanFactory);
        
        try {
            // 4. 后置处理 BeanFactory
            postProcessBeanFactory(beanFactory);
            
            // 5. 调用 BeanFactory 后置处理器
            invokeBeanFactoryPostProcessors(beanFactory);
            
            // 6. 注册 Bean 后置处理器
            registerBeanPostProcessors();
            
            // 7. 初始化消息源
            initMessageSource();
            
            // 8. 初始化事件多播器
            initApplicationEventMulticaster();
            
            // 9. 留给子类的模板方法
            onRefresh();
            
            // 10. 注册事件监听器
            registerListeners();
            
            // 11. 实例化剩余的单例 Bean
            finishBeanFactoryInitialization(beanFactory);
            
            // 12. 完成刷新
            finishRefresh();
        }
        catch (Exception ex) {
            destroyBeans();
            throw new RuntimeException("Failed to refresh context", ex);
        }
    }
    
    protected void prepareRefresh() {
        System.out.println("准备刷新容器...");
    }
    
    protected DefaultListableBeanFactory obtainFreshBeanFactory() {
        return this;
    }
    
    protected void prepareBeanFactory(DefaultListableBeanFactory beanFactory) {
        // 添加默认的 BeanPostProcessor 等
    }
    
    protected void postProcessBeanFactory(DefaultListableBeanFactory beanFactory) {
    }
    
    protected void invokeBeanFactoryPostProcessors(DefaultListableBeanFactory beanFactory) {
    }
    
    protected void registerBeanPostProcessors() {
    }
    
    protected void initMessageSource() {
    }
    
    protected void initApplicationEventMulticaster() {
    }
    
    protected void onRefresh() {
    }
    
    protected void registerListeners() {
    }
    
    protected void finishBeanFactoryInitialization(DefaultListableBeanFactory beanFactory) {
        // 实例化所有非懒加载的单例 Bean
        String[] beanNames = beanFactory.getBeanDefinitionNames();
        for (String beanName : beanNames) {
            beanFactory.getBean(beanName);
        }
    }
    
    protected void finishRefresh() {
        System.out.println("容器刷新完成");
    }
    
    protected void destroyBeans() {
        close();
    }
    
    @Override
    public void close() {
        System.out.println("关闭容器...");
        destroySingletons();
    }
    
    @Override
    public void registerShutdownHook() {
        Runtime.getRuntime().addShutdownHook(new Thread(() -> {
            close();
        }));
    }
    
    @Override
    public String[] getBeanDefinitionNames() {
        return getBeanDefinitionNames();
    }
}
```

## 三、AnnotationConfigApplicationContext

支持注解配置的应用上下文。

```java
package com.example.minispring.context;

import com.example.minispring.beans.config.BeanDefinition;
import com.example.minispring.beans.factory.support.DefaultListableBeanFactory;
import com.example.minispring.context.support.AbstractApplicationContext;

import java.util.Set;

/**
 * 注解配置应用上下文
 */
public class AnnotationConfigApplicationContext 
        extends AbstractApplicationContext {
    
    private final String basePackage;
    
    public AnnotationConfigApplicationContext(String basePackage) {
        this.basePackage = basePackage;
        refresh();
    }
    
    @Override
    protected void onRefresh() {
        // 扫描包，注册 Bean 定义
        scanPackages();
    }
    
    private void scanPackages() {
        // 扫描指定包下的 @Component、@Service、@Controller 等注解
        // 这里简化实现，实际需要使用 ClassPathScanningCandidateComponentProvider
        
        System.out.println("扫描包：" + basePackage);
        
        // 示例：手动注册 Bean 定义
        // 实际应该通过扫描获取
    }
    
    @Override
    protected void finishBeanFactoryInitialization(DefaultListableBeanFactory beanFactory) {
        super.finishBeanFactoryInitialization(beanFactory);
    }
}
```

## 四、完整使用示例

```java
package com.example.minispring.test;

import com.example.minispring.beans.annotation.Autowired;
import com.example.minispring.beans.annotation.Component;
import com.example.minispring.beans.factory.InitializingBean;
import com.example.minispring.beans.factory.support.AutowiredAnnotationBeanPostProcessor;
import com.example.minispring.context.AnnotationConfigApplicationContext;

public class MiniSpringDemo {
    
    public static void main(String[] args) {
        // 1. 创建应用上下文
        AnnotationConfigApplicationContext context = 
            new AnnotationConfigApplicationContext("com.example");
        
        // 2. 注册 @Autowired 处理器
        context.addBeanPostProcessor(new AutowiredAnnotationBeanPostProcessor(context));
        
        // 3. 获取 Bean
        UserService userService = context.getBean(UserService.class);
        
        // 4. 使用 Bean
        userService.createUser();
        
        // 5. 关闭上下文
        context.close();
    }
}

@Component
class UserService implements InitializingBean {
    
    @Autowired
    private UserDao userDao;
    
    @Override
    public void afterPropertiesSet() {
        System.out.println("UserService 初始化完成");
    }
    
    public void createUser() {
        userDao.insert();
        System.out.println("用户创建成功");
    }
    
    public UserDao getUserDao() {
        return userDao;
    }
}

@Component
class UserDao {
    
    public void insert() {
        System.out.println("插入用户数据");
    }
}
```

```
扫描包：com.example
UserService 初始化完成
容器刷新完成
插入用户数据
用户创建成功
关闭容器...
```

## 六、小结

1. **ApplicationContext**：扩展 BeanFactory，提供更完整的功能

4. **完整流程**：扫描 → 注册 → 实例化 → 初始化 → 使用 → 销毁
