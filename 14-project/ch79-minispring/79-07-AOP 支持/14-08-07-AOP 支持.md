# 79.7 AOP 支持

## 一、ProxyFactory 代理工厂

创建 AOP 代理的工厂类。

```java
package com.example.minispring.aop;

import java.lang.reflect.Proxy;

/**
 * 代理工厂：创建 AOP 代理
 */
public class ProxyFactory {
    
    private Object target;
    private Advice advice;
    
    public ProxyFactory(Object target) {
        this.target = target;
    }
    
    public void addAdvice(Advice advice) {
        this.advice = advice;
    }
    
    /**
     * 获取代理对象
     */
    public Object getProxy() {
        return Proxy.newProxyInstance(
            target.getClass().getClassLoader(),
            target.getClass().getInterfaces(),
            (proxy, method, args) -> {
                // 执行环绕通知
                if (advice instanceof MethodInterceptor) {
                    return ((MethodInterceptor) advice).invoke(
                        new MethodInvocation(target, method, args)
                    );
                }
                return method.invoke(target, args);
            }
        );
    }
}
```

## 二、通知接口

```java
package com.example.minispring.aop;

/**
 * 通知接口
 */
public interface Advice {
}

/**
 * 方法拦截器（环绕通知）
 */
public interface MethodInterceptor extends Advice {
    
    /**
     * 拦截方法调用
     */
    Object invoke(MethodInvocation invocation) throws Throwable;
}

/**
 * 方法调用
 */
public class MethodInvocation {
    
    private Object target;
    private java.lang.reflect.Method method;
    private Object[] args;
    
    public MethodInvocation(Object target, java.lang.reflect.Method method, Object[] args) {
        this.target = target;
        this.method = method;
        this.args = args;
    }
    
    /**
     * 执行目标方法
     */
    public Object proceed() throws Exception {
        return method.invoke(target, args);
    }
    
    public java.lang.reflect.Method getMethod() {
        return method;
    }
    
    public Object[] getArguments() {
        return args;
    }
}
```

## 三、前置通知

```java
package com.example.minispring.aop;

/**
 * 前置通知
 */
public interface BeforeAdvice extends Advice {
}

/**
 * 方法执行前通知
 */
public interface MethodBeforeAdvice extends BeforeAdvice {
    
    /**
     * 方法执行前执行
     */
    void before(java.lang.reflect.Method method, Object[] args, Object target) throws Throwable;
}

/**
 * 前置通知拦截器
 */
public class BeforeAdviceInterceptor implements MethodInterceptor {
    
    private MethodBeforeAdvice advice;
    
    public BeforeAdviceInterceptor(MethodBeforeAdvice advice) {
        this.advice = advice;
    }
    
    @Override
    public Object invoke(MethodInvocation invocation) throws Throwable {
        // 执行前置通知
        advice.before(invocation.getMethod(), invocation.getArguments(), invocation.getTarget());
        
        // 执行目标方法
        return invocation.proceed();
    }
}
```

## 四、环绕通知示例

```java
package com.example.minispring.test;

import com.example.minispring.aop.MethodInterceptor;
import com.example.minispring.aop.MethodInvocation;
import com.example.minispring.aop.ProxyFactory;

public class AopTest {
    
    public static void main(String[] args) {
        // 1. 创建目标对象
        UserService target = new UserService();
        
        // 2. 创建代理工厂
        ProxyFactory proxyFactory = new ProxyFactory(target);
        
        // 3. 添加环绕通知
        proxyFactory.addAdvice(new MethodInterceptor() {
            @Override
            public Object invoke(MethodInvocation invocation) throws Throwable {
                System.out.println("[环绕通知] 方法执行前");
                
                long start = System.currentTimeMillis();
                
                // 执行目标方法
                Object result = invocation.proceed();
                
                long end = System.currentTimeMillis();
                System.out.println("[环绕通知] 方法执行后，耗时：" + (end - start) + "ms");
                
                return result;
            }
        });
        
        // 4. 获取代理对象
        UserService proxy = (UserService) proxyFactory.getProxy();
        
        // 5. 通过代理调用方法
        proxy.sayHello();
    }
}

interface UserService {
    void sayHello();
}

class UserServiceImpl implements UserService {
    @Override
    public void sayHello() {
        System.out.println("Hello!");
        try {
            Thread.sleep(100);
        } catch (InterruptedException e) {}
    }
}
```

**输出**：
```
[环绕通知] 方法执行前
Hello!
[环绕通知] 方法执行后，耗时：100ms
```

## 五、切点匹配

```java
package com.example.minispring.aop;

import java.lang.reflect.Method;

/**
 * 切点接口
 */
public interface Pointcut {
    
    /**
     * 获取方法匹配器
     */
    MethodMatcher getMethodMatcher();
}

/**
 * 方法匹配器接口
 */
public interface MethodMatcher {
    
    /**
     * 是否匹配指定方法
     */
    boolean matches(Method method, Class<?> targetClass);
}

/**
 * 切面信息
 */
public class AspectExpression {
    
    private String expression;
    
    public AspectExpression(String expression) {
        this.expression = expression;
    }
    
    public String getExpression() {
        return expression;
    }
}
```

## 六、AdvisedSupport 通知支持

```java
package com.example.minispring.aop;

import java.util.ArrayList;
import java.util.List;

/**
 * 通知支持类
 */
public class AdvisedSupport {
    
    /** 目标对象 */
    private Object target;
    
    /** 通知列表 */
    private final List<MethodInterceptor> interceptors = new ArrayList<>();
    
    /** 切点 */
    private Pointcut pointcut;
    
    public void setTarget(Object target) {
        this.target = target;
    }
    
    public Object getTarget() {
        return target;
    }
    
    public void addInterceptor(MethodInterceptor interceptor) {
        interceptors.add(interceptor);
    }
    
    public List<MethodInterceptor> getInterceptors() {
        return interceptors;
    }
    
    public void setPointcut(Pointcut pointcut) {
        this.pointcut = pointcut;
    }
    
    public Pointcut getPointcut() {
        return pointcut;
    }
}
```

## 七、小结

1. **ProxyFactory**：代理工厂，创建 AOP 代理
2. **Advice**：通知接口（前置、后置、环绕）
3. **MethodInterceptor**：方法拦截器（环绕通知）
4. **Pointcut**：切点，匹配方法
5. **AdvisedSupport**：通知支持类
