# 79.5 属性填充（依赖注入）

## 一、属性填充实现

实现依赖注入，将 Bean 的属性值填充到 Bean 实例中。

```java
package com.example.minispring.beans.factory.support;

import com.example.minispring.beans.config.BeanDefinition;
import com.example.minispring.beans.config.PropertyValue;
import com.example.minispring.beans.config.PropertyValues;
import com.example.minispring.beans.exception.BeansException;

import java.lang.reflect.Field;

/**
 * 属性填充：实现依赖注入
 */
public abstract class AbstractAutowireCapableBeanFactory extends AbstractBeanFactory {
    
    // ... 前面的代码
    
    /**
     * 属性填充
     */
    protected void populateBean(String beanName, BeanDefinition bd, Object bean) {
        // 1. 获取定义的属性值
        PropertyValues pvs = bd.getPropertyValues();
        
        // 2. 应用属性值
        for (PropertyValue pv : pvs.getPropertyValues()) {
            String propertyName = pv.getName();
            Object value = pv.getValue();
            
            // 3. 设置属性
            try {
                Field field = bean.getClass().getDeclaredField(propertyName);
                field.setAccessible(true);
                field.set(bean, value);
            } catch (Exception e) {
                throw new BeansException("Failed to set property [" + propertyName + "]", e);
            }
        }
    }
}
```

## 二、BeanPostProcessor 接口

Bean 后置处理器，在 Bean 初始化前后执行自定义逻辑。

```java
package com.example.minispring.beans.factory.config;

import com.example.minispring.beans.exception.BeansException;

/**
 * Bean 后置处理器接口
 */
public interface BeanPostProcessor {
    
    /**
     * 在 Bean 初始化之前执行
     */
    Object postProcessBeforeInitialization(Object bean, String beanName) throws BeansException;
    
    /**
     * 在 Bean 初始化之后执行
     */
    Object postProcessAfterInitialization(Object bean, String beanName) throws BeansException;
}
```

## 三、AutowiredAnnotationBeanPostProcessor

处理 @Autowired 注解的 Bean 后置处理器。

```java
package com.example.minispring.beans.factory.support;

import com.example.minispring.beans.annotation.Autowired;
import com.example.minispring.beans.exception.BeansException;
import com.example.minispring.beans.factory.config.BeanPostProcessor;

import java.lang.reflect.Field;

/**
 * @Autowired 注解处理器
 */
public class AutowiredAnnotationBeanPostProcessor implements BeanPostProcessor {
    
    private BeanFactory beanFactory;
    
    public AutowiredAnnotationBeanPostProcessor(BeanFactory beanFactory) {
        this.beanFactory = beanFactory;
    }
    
    @Override
    public Object postProcessBeforeInitialization(Object bean, String beanName) {
        // 处理 @Autowired 注解（字段注入）
        Class<?> beanClass = bean.getClass();
        
        for (Field field : beanClass.getDeclaredFields()) {
            if (field.isAnnotationPresent(Autowired.class)) {
                field.setAccessible(true);
                String fieldName = field.getName();
                
                try {
                    // 从 BeanFactory 获取依赖的 Bean
                    Object fieldValue = beanFactory.getBean(fieldName);
                    field.set(bean, fieldValue);
                } catch (BeansException e) {
                    Autowired autowired = field.getAnnotation(Autowired.class);
                    if (autowired.required()) {
                        throw e;
                    }
                }
            }
        }
        
        return bean;
    }
    
    @Override
    public Object postProcessAfterInitialization(Object bean, String beanName) {
        return bean;
    }
}
```

## 四、@Autowired 注解

```java
package com.example.minispring.beans.annotation;

import java.lang.annotation.*;

@Target({ElementType.FIELD, ElementType.METHOD, ElementType.PARAMETER})
@Retention(RetentionPolicy.RUNTIME)
@Documented
public @interface Autowired {
    
    /**
     * 是否必需
     */
    boolean required() default true;
}
```

## 五、整合 BeanPostProcessor

在 Bean 创建流程中调用 BeanPostProcessor。

```java
package com.example.minispring.beans.factory.support;

import com.example.minispring.beans.config.BeanDefinition;
import com.example.minispring.beans.exception.BeansException;
import com.example.minispring.beans.factory.config.BeanPostProcessor;

import java.util.ArrayList;
import java.util.List;

public abstract class AbstractAutowireCapableBeanFactory extends AbstractBeanFactory {
    
    /** BeanPostProcessor 列表 */
    private final List<BeanPostProcessor> beanPostProcessors = new ArrayList<>();
    
    /**
     * 添加 BeanPostProcessor
     */
    public void addBeanPostProcessor(BeanPostProcessor beanPostProcessor) {
        beanPostProcessors.add(beanPostProcessor);
    }
    
    @Override
    protected Object createBean(String beanName, BeanDefinition bd) {
        // 1. 实例化 Bean
        Object bean = instantiateBean(beanName, bd);
        
        // 2. 属性填充
        populateBean(beanName, bd, bean);
        
        // 3. 执行 BeanPostProcessor 前置处理
        for (BeanPostProcessor bpp : beanPostProcessors) {
            bean = bpp.postProcessBeforeInitialization(bean, beanName);
        }
        
        // 4. 初始化 Bean
        initializeBean(beanName, bean, bd);
        
        // 5. 执行 BeanPostProcessor 后置处理
        for (BeanPostProcessor bpp : beanPostProcessors) {
            bean = bpp.postProcessAfterInitialization(bean, beanName);
        }
        
        return bean;
    }
}
```

## 六、测试代码

```java
package com.example.minispring.test;

import com.example.minispring.beans.annotation.Autowired;
import com.example.minispring.beans.config.BeanDefinition;
import com.example.minispring.beans.config.PropertyValue;
import com.example.minispring.beans.factory.support.AutowiredAnnotationBeanPostProcessor;
import com.example.minispring.beans.factory.support.DefaultListableBeanFactory;

public class DependencyInjectionTest {
    
    public static void main(String[] args) {
        // 1. 创建 Bean 工厂
        DefaultListableBeanFactory factory = new DefaultListableBeanFactory();
        
        // 2. 注册 UserDao
        BeanDefinition userDaoBd = new BeanDefinition(UserDao.class);
        factory.registerBeanDefinition("userDao", userDaoBd);
        
        // 3. 注册 UserService
        BeanDefinition userServiceBd = new BeanDefinition(UserService.class);
        userServiceBd.getPropertyValues().addPropertyValue(
            new PropertyValue("userDao", null) // 值不重要，会被 @Autowired 覆盖
        );
        factory.registerBeanDefinition("userService", userServiceBd);
        
        // 4. 注册 @Autowired 处理器
        factory.addBeanPostProcessor(new AutowiredAnnotationBeanPostProcessor(factory));
        
        // 5. 获取 UserService（会自动注入 UserDao）
        UserService userService = (UserService) factory.getBean("userService");
        
        // 6. 使用 Bean
        userService.createUser();
        
        // 7. 验证注入
        System.out.println("UserDao 是否注入：" + (userService.getUserDao() != null));
    }
}

class UserDao {
    public void insert() {
        System.out.println("插入用户数据");
    }
}

class UserService {
    
    @Autowired
    private UserDao userDao;
    
    public UserDao getUserDao() {
        return userDao;
    }
    
    public void createUser() {
        userDao.insert();
        System.out.println("用户创建成功");
    }
}
```

**输出**：
```
插入用户数据
用户创建成功
UserDao 是否注入：true
```

## 七、小结

1. **populateBean**：属性填充实现依赖注入
2. **BeanPostProcessor**：Bean 后置处理器接口
3. **AutowiredAnnotationBeanPostProcessor**：处理 @Autowired 注解
4. **@Autowired**：自动注入注解
