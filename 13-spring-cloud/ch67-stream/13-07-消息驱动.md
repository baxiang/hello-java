# 13.7 消息驱动

## 一、Spring Cloud Stream 概述

Spring Cloud Stream 是构建消息驱动微服务的框架，屏蔽了底层消息中间件的差异。

**核心概念**：
- Binder：消息中间件抽象
- Channel：消息通道
- Source：消息生产者
- Sink：消息消费者
- Processor：既是生产者也是消费者

## 二、添加依赖

```xml
<!-- RabbitMQ Binder -->
<dependency>
    <groupId>org.springframework.cloud</groupId>
    <artifactId>spring-cloud-starter-stream-rabbit</artifactId>
</dependency>

<!-- 或 Kafka Binder -->
<dependency>
    <groupId>org.springframework.cloud</groupId>
    <artifactId>spring-cloud-starter-stream-kafka</artifactId>
</dependency>
```

## 三、基本使用

### 3.1 配置文件

```yaml
spring:
  cloud:
    stream:
      bindings:
        # 输出通道
        output:
          destination: order-topic
          content-type: application/json
        # 输入通道
        input:
          destination: order-topic
          content-type: application/json
          group: order-group  # 消费者组
      binders:
        defaultRabbit:
          type: rabbit
          environment:
            spring:
              rabbitmq:
                host: localhost
                port: 5672
                username: guest
                password: guest
```

### 3.2 消息生产者

```java
@EnableBinding(Source.class)
public class OrderProducer {
    
    @Autowired
    private Source source;
    
    public void sendOrder(Order order) {
        source.output().send(MessageBuilder.withPayload(order).build());
    }
}

// 使用
@RestController
@RequestMapping("/api/orders")
public class OrderController {
    
    @Autowired
    private OrderProducer orderProducer;
    
    @PostMapping
    public Order create(@RequestBody Order order) {
        // 保存订单
        orderService.save(order);
        
        // 发送消息
        orderProducer.sendOrder(order);
        
        return order;
    }
}
```

### 3.3 消息消费者

```java
@EnableBinding(Sink.class)
public class OrderConsumer {
    
    @StreamListener(Sink.INPUT)
    public void handleOrder(Order order) {
        System.out.println("收到订单：" + order);
        
        // 处理订单
        // 发送通知、更新库存等
    }
}
```

## 四、自定义通道

### 4.1 定义通道接口

```java
public interface OrderChannels {
    
    String ORDER_OUTPUT = "orderOutput";
    String ORDER_INPUT = "orderInput";
    
    @Output(OrderChannels.ORDER_OUTPUT)
    MessageChannel orderOutput();
    
    @Input(OrderChannels.ORDER_INPUT)
    SubscribableChannel orderInput();
}
```

### 4.2 生产者配置

```java
@EnableBinding(OrderChannels.class)
public class OrderProducer {
    
    @Autowired
    private OrderChannels channels;
    
    public void sendOrder(Order order) {
        channels.orderOutput().send(
            MessageBuilder.withPayload(order).build()
        );
    }
    
    public void sendPayment(Payment payment) {
        channels.orderOutput().send(
            MessageBuilder.withPayload(payment).build()
        );
    }
}
```

### 4.3 消费者配置

```java
@EnableBinding(OrderChannels.class)
public class OrderConsumer {
    
    @StreamListener(OrderChannels.ORDER_INPUT)
    public void handleOrder(Order order) {
        System.out.println("处理订单：" + order.getId());
    }
    
    @StreamListener(OrderChannels.ORDER_INPUT)
    public void handlePayment(Payment payment) {
        System.out.println("处理支付：" + payment.getId());
    }
}
```

## 五、消息分区

### 5.1 生产者配置

```yaml
spring:
  cloud:
    stream:
      bindings:
        output:
          destination: partitioned-topic
          producer:
            partition-key-expression: payload.userId  # 分区键
            partition-count: 3  # 分区数
```

### 5.2 消费者配置

```yaml
spring:
  cloud:
    stream:
      bindings:
        input:
          destination: partitioned-topic
          group: order-group
          consumer:
            partitioned: true  # 启用分区
      instance-count: 3  # 实例数
      instance-index: 0  # 当前实例索引（0, 1, 2）
```

## 六、错误处理

### 6.1 重试配置

```yaml
spring:
  cloud:
    stream:
      bindings:
        input:
          consumer:
            max-attempts: 3  # 最大重试次数
            back-off-initial-interval: 1000  # 初始间隔
            back-off-multiplier: 2  # 倍数
            back-off-max-interval: 10000  # 最大间隔
```

### 6.2 死信队列

```yaml
spring:
  cloud:
    stream:
      bindings:
        input:
          destination: order-topic
          group: order-group
        input.DLQ:
          destination: order-topic.DLQ  # 死信队列
      rabbit:
        bindings:
          input:
            consumer:
              auto-bind-dlq: true  # 自动绑定死信队列
              republish-to-dlq: true  # 重新发布到死信队列
```

### 6.3 错误处理器

```java
@EnableBinding(Sink.class)
public class OrderConsumer {
    
    @StreamListener(Sink.INPUT)
    public void handleOrder(Order order) {
        if (order.getAmount().compareTo(BigDecimal.ZERO) <= 0) {
            throw new IllegalArgumentException("金额必须大于 0");
        }
        // 处理订单
    }
    
    // 错误处理
    @StreamListener(Sink.INPUT)
    @Retryable(maxAttempts = 3, backoff = @Backoff(delay = 1000))
    public void handleOrderWithRetry(Order order) {
        // 处理订单
    }
    
    // 恢复处理
    @StreamListener(Sink.INPUT)
    @Recover
    public void recover(IllegalArgumentException e, Order order) {
        System.out.println("重试失败，订单：" + order.getId());
        // 记录日志、发送告警等
    }
}
```

## 七、综合示例

### 7.1 订单事件处理

```java
// 通道定义
public interface OrderEventChannels {
    
    String ORDER_CREATED = "orderCreated";
    String ORDER_PAID = "orderPaid";
    String ORDER_SHIPPED = "orderShipped";
    
    @Output(OrderEventChannels.ORDER_CREATED)
    MessageChannel orderCreated();
    
    @Output(OrderEventChannels.ORDER_PAID)
    MessageChannel orderPaid();
    
    @Output(OrderEventChannels.ORDER_SHIPPED)
    MessageChannel orderShipped();
    
    @Input(OrderEventChannels.ORDER_CREATED)
    SubscribableChannel orderCreatedInput();
    
    @Input(OrderEventChannels.ORDER_PAID)
    SubscribableChannel orderPaidInput();
    
    @Input(OrderEventChannels.ORDER_SHIPPED)
    SubscribableChannel orderShippedInput();
}

// 事件生产者
@EnableBinding(OrderEventChannels.class)
public class OrderEventProducer {
    
    @Autowired
    private OrderEventChannels channels;
    
    public void sendOrderCreated(Order order) {
        channels.orderCreated().send(
            MessageBuilder.withPayload(order).build()
        );
    }
    
    public void sendOrderPaid(Order order) {
        channels.orderPaid().send(
            MessageBuilder.withPayload(order).build()
        );
    }
    
    public void sendOrderShipped(Order order) {
        channels.orderShipped().send(
            MessageBuilder.withPayload(order).build()
        );
    }
}

// 事件消费者
@EnableBinding(OrderEventChannels.class)
public class OrderEventConsumer {
    
    @Autowired
    private InventoryService inventoryService;
    
    @Autowired
    private NotificationService notificationService;
    
    // 订单创建 - 扣减库存
    @StreamListener(OrderEventChannels.ORDER_CREATED)
    public void handleOrderCreated(Order order) {
        inventoryService.decreaseStock(order.getItems());
    }
    
    // 订单支付 - 发送通知
    @StreamListener(OrderEventChannels.ORDER_PAID)
    public void handleOrderPaid(Order order) {
        notificationService.sendOrderPaidNotification(order);
    }
    
    // 订单发货 - 发送通知
    @StreamListener(OrderEventChannels.ORDER_SHIPPED)
    public void handleOrderShipped(Order order) {
        notificationService.sendOrderShippedNotification(order);
    }
}
```

### 7.2 完整配置

```yaml
spring:
  cloud:
    stream:
      binders:
        defaultRabbit:
          type: rabbit
          environment:
            spring:
              rabbitmq:
                host: localhost
                port: 5672
                username: guest
                password: guest
                listener:
                  simple:
                    acknowledge-mode: manual  # 手动确认
      bindings:
        orderCreated:
          destination: order-created
          content-type: application/json
          group: order-service
          producer:
            required-groups: inventory,notification
        orderPaid:
          destination: order-paid
          content-type: application/json
          group: order-service
        orderShipped:
          destination: order-shipped
          content-type: application/json
          group: order-service
        orderCreatedInput:
          destination: order-created
          content-type: application/json
          group: inventory-service
          consumer:
            max-attempts: 3
        orderPaidInput:
          destination: order-paid
          content-type: application/json
          group: notification-service
        orderShippedInput:
          destination: order-shipped
          content-type: application/json
          group: notification-service
```
