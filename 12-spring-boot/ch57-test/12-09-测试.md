## 一、JUnit 5

### 1.1 添加依赖

```xml
<dependency>

    <scope>test</scope>
</dependency>
```

### 1.2 基本测试

```java
@SpringBootTest
class DemoApplicationTests {
    
    @Test
    void contextLoads() {
        // 验证应用可以启动
    }
    
    @Test
    void testAdd() {
        int result = 2 + 3;
        assertEquals(5, result);
    }
    
    @Test
    void testList() {
        List<String> list = Arrays.asList("A", "B", "C");
        assertTrue(list.contains("B"));
        assertEquals(3, list.size());
    }
}
```

### 1.3 常用断言

```java
@SpringBootTest
class AssertTests {
    
    @Test
    void testAssertions() {
        // 等于
        assertEquals(5, 2 + 3);
        assertEquals("hello", "hello");
        
        // 不等于
        assertNotEquals(5, 2 + 4);
        
        // 真/假
        assertTrue(2 > 1);
        assertFalse(2 < 1);
        
        // 空值
        assertNull(null);
        assertNotNull("not null");
        
        // 数组
        assertArrayEquals(new int[]{1, 2, 3}, new int[]{1, 2, 3});
        
        // 异常
        assertThrows(IllegalArgumentException.class, () -> {
            throw new IllegalArgumentException("error");
        });
        
        // 所有断言
        assertAll("group",
            () -> assertEquals(1, 1),
            () -> assertEquals(2, 2)
        );
    }
}
```

## 二、MockMvc

### 2.1 Controller 测试

```java
@WebMvcTest(UserController.class)
class UserControllerTest {
    
    @Autowired
    private MockMvc mockMvc;
    
    @MockBean
    private UserService userService;
    
    @Test
    void testGetUser() throws Exception {
        // 准备数据
        User user = new User(1L, "张三", 18);
        when(userService.findById(1L)).thenReturn(user);
        
        // 执行请求并验证
        mockMvc.perform(get("/api/users/1"))
            .andExpect(status().isOk())
            .andExpect(jsonPath("$.code").value(200))
            .andExpect(jsonPath("$.data.id").value(1))
            .andExpect(jsonPath("$.data.name").value("张三"));
    }
    
    @Test
    void testCreateUser() throws Exception {
        User user = new User(1L, "张三", 18);
        when(userService.save(any(User.class))).thenReturn(user);
        
        mockMvc.perform(post("/api/users")
                .contentType(MediaType.APPLICATION_JSON)
                .content("{\"name\":\"张三\",\"age\":18}"))
            .andExpect(status().isOk())
            .andExpect(jsonPath("$.code").value(200));
    }
    
    @Test
    void testDeleteUser() throws Exception {
        doNothing().when(userService).delete(1L);
        
        mockMvc.perform(delete("/api/users/1"))
            .andExpect(status().isOk());
    }
}
```

### 2.2 测试认证

## 三、Mockito

### 3.1 基本用法

```java
@SpringBootTest
class UserServiceTest {
    
    @Autowired
    private UserService userService;
    
    @MockBean
    private UserMapper userMapper;
    
    @Test
    void testFindById() {
        // 准备数据
        User mockUser = new User(1L, "张三", 18);
        when(userMapper.findById(1L)).thenReturn(mockUser);
        
        // 执行测试
        User result = userService.findById(1L);
        
        // 验证结果
        assertNotNull(result);
        assertEquals("张三", result.getName());
        
        // 验证调用
        verify(userMapper, times(1)).findById(1L);
    }
    
    @Test
    void testSave() {
        User newUser = new User(null, "李四", 20);
        User savedUser = new User(1L, "李四", 20);
        
        when(userMapper.insert(newUser)).thenReturn(savedUser);
        
        User result = userService.save(newUser);
        
        assertNotNull(result.getId());
        verify(userMapper).insert(newUser);
    }
    
    @Test
    void testDelete() {
        doNothing().when(userMapper).delete(1L);
        
        userService.delete(1L);
        
        verify(userMapper).delete(1L);
    }
}
```

### 3.2 参数匹配

```java
@Test
void testArgumentMatchers() {
    // 任意字符串
    when(userMapper.findByName(anyString())).thenReturn(new User());
    
    // 任意 Long
    when(userMapper.findById(anyLong())).thenReturn(new User());
    
    // 包含特定字符串
    when(userMapper.findByName(contains("张"))).thenReturn(new User());
    
    // 正则匹配
    when(userMapper.findByName(matches("^张.*"))).thenReturn(new User());
    
    // 多个参数
    when(userMapper.findByNameAndAge(anyString(), anyInt()))
        .thenReturn(new User());
}
```

### 3.3 验证调用

```java
@Test
void testVerification() {
    userService.process(1L);
    
    // 验证调用次数
    verify(userMapper, times(1)).findById(1L);
    verify(userMapper, never()).delete(1L);
    verify(userMapper, atLeastOnce()).findAll();
    
    // 验证调用顺序
    InOrder inOrder = inOrder(userMapper);
    inOrder.verify(userMapper).findById(1L);
    inOrder.verify(userMapper).update(any());
}
```

## 四、集成测试

### 4.1 数据库测试

```java
@SpringBootTest
@AutoConfigureTestDatabase(replace = AutoConfigureTestDatabase.Replace.NONE)
class UserRepositoryTest {
    
    @Autowired
    private UserRepository userRepository;
    
    @Test
    void testSave() {
        User user = new User();
        user.setUsername("test");
        user.setPassword("123456");
        
        User saved = userRepository.save(user);
        
        assertNotNull(saved.getId());
        assertEquals("test", saved.getUsername());
    }
    
    @Test
    void testFind() {
        User user = new User();
        user.setUsername("findtest");
        user.setPassword("123456");
        userRepository.save(user);
        
        Optional<User> found = userRepository.findByUsername("findtest");
        
        assertTrue(found.isPresent());
        assertEquals("findtest", found.get().getUsername());
    }
    
    @Test
    void testDelete() {
        User user = new User();
        user.setUsername("deletetest");
        user.setPassword("123456");
        userRepository.save(user);
        
        userRepository.delete(user);
        
        Optional<User> found = userRepository.findByUsername("deletetest");
        assertFalse(found.isPresent());
    }
}
```

## 六、综合示例

### 6.1 完整测试类

```java
@WebMvcTest(OrderController.class)
class OrderControllerTest {
    
    @Autowired
    private MockMvc mockMvc;
    
    @MockBean
    private OrderService orderService;
    
    @Autowired
    private ObjectMapper objectMapper;
    
    @WithMockUser(username = "user1", roles = "USER")
    @Test
    void testCreateOrder() throws Exception {
        // 准备数据
        OrderDTO dto = new OrderDTO();
        dto.setProductId(1L);
        dto.setQuantity(2);
        
        Order order = new Order();
        order.setId(1L);
        order.setUserId(1L);
        order.setAmount(new BigDecimal("200.00"));
        
        when(orderService.create(any(), any())).thenReturn(order);
        
        // 执行请求
        mockMvc.perform(post("/api/orders")
                .contentType(MediaType.APPLICATION_JSON)
                .content(objectMapper.writeValueAsString(dto)))
            .andExpect(status().isOk())
            .andExpect(jsonPath("$.code").value(200))
            .andExpect(jsonPath("$.data.id").value(1))
            .andExpect(jsonPath("$.data.amount").value(200.00));
    }
    
    @WithMockUser(username = "user1", roles = "USER")
    @Test
    void testGetOrder() throws Exception {
        Order order = new Order();
        order.setId(1L);
        order.setUserId(1L);
        order.setAmount(new BigDecimal("200.00"));
        
        when(orderService.findById(1L)).thenReturn(order);
        
        mockMvc.perform(get("/api/orders/1"))
            .andExpect(status().isOk())
            .andExpect(jsonPath("$.data.id").value(1));
    }
    
    @WithMockUser(username = "user1", roles = "USER")
    @Test
    void testGetOrders() throws Exception {
        List<Order> orders = Arrays.asList(
            new Order(1L, 1L, new BigDecimal("100.00")),
            new Order(2L, 1L, new BigDecimal("200.00"))
        );
        
        when(orderService.findAll(any())).thenReturn(orders);
        
        mockMvc.perform(get("/api/orders"))
            .andExpect(status().isOk())
            .andExpect(jsonPath("$.data.length()").value(2));
    }
}
```

### 6.2 Service 层测试

```java
@ExtendWith(MockitoExtension.class)
class OrderServiceTest {
    
    @InjectMocks
    private OrderService orderService;
    
    @Mock
    private OrderMapper orderMapper;
    
    @Mock
    private ProductMapper productMapper;
    
    @Test
    void testCreateOrder() {
        // 准备数据
        OrderDTO dto = new OrderDTO();
        dto.setProductId(1L);
        dto.setQuantity(2);
        
        Product product = new Product();
        product.setId(1L);
        product.setPrice(new BigDecimal("100.00"));
        product.setStock(10);
        
        Order savedOrder = new Order();
        savedOrder.setId(1L);
        savedOrder.setAmount(new BigDecimal("200.00"));
        
        // 设置 Mock
        when(productMapper.findById(1L)).thenReturn(product);
        when(orderMapper.insert(any())).thenReturn(savedOrder);
        
        // 执行测试
        Order result = orderService.create(dto, 1L);
        
        // 验证结果
        assertNotNull(result);
        assertEquals(new BigDecimal("200.00"), result.getAmount());
        
        // 验证调用
        verify(productMapper).findById(1L);
        verify(orderMapper).insert(any());
    }
}
```
