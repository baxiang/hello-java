# 8.6 继承综合案例

## 一、案例一：电商系统

### 1.1 商品类设计

```java
// 父类：商品
class Product {
    protected String id;
    protected String name;
    protected double price;
    protected int stock;
    
    public Product(String id, String name, double price, int stock) {
        this.id = id;
        this.name = name;
        setPrice(price);
        setStock(stock);
    }
    
    public void setPrice(double price) {
        if (price > 0) {
            this.price = price;
        }
    }
    
    public void setStock(int stock) {
        if (stock >= 0) {
            this.stock = stock;
        }
    }
    
    public double getPrice() {
        return price;
    }
    
    public int getStock() {
        return stock;
    }
    
    public double getTotalPrice() {
        return price * stock;
    }
    
    @Override
    public String toString() {
        return "Product{id='" + id + "', name='" + name + 
               "', price=" + price + ", stock=" + stock + "}";
    }
}

// 子类：实体商品
class PhysicalProduct extends Product {
    private double weight;
    private String shippingFrom;
    
    public PhysicalProduct(String id, String name, double price, int stock, 
                          double weight, String shippingFrom) {
        super(id, name, price, stock);
        this.weight = weight;
        this.shippingFrom = shippingFrom;
    }
    
    public double getWeight() {
        return weight;
    }
    
    public double getShippingFee() {
        return weight * 2 + 5;  // 运费计算
    }
    
    @Override
    public double getTotalPrice() {
        return super.getTotalPrice() + getShippingFee();
    }
    
    @Override
    public String toString() {
        return "PhysicalProduct{" +
                "id='" + id + "', name='" + name + '\'' +
                ", price=" + price +
                ", weight=" + weight + "kg" +
                ", shippingFee=" + getShippingFee() +
                '}';
    }
}

// 子类：虚拟商品
class DigitalProduct extends Product {
    private String downloadUrl;
    private int validDays;
    
    public DigitalProduct(String id, String name, double price, int stock,
                         String downloadUrl, int validDays) {
        super(id, name, price, stock);
        this.downloadUrl = downloadUrl;
        this.validDays = validDays;
    }
    
    public void download() {
        System.out.println("从 " + downloadUrl + " 下载");
    }
    
    @Override
    public String toString() {
        return "DigitalProduct{" +
                "id='" + id + "', name='" + name + '\'' +
                ", price=" + price +
                ", validDays=" + validDays + "天" +
                '}';
    }
}

// 子类：打折商品
class DiscountProduct extends Product {
    private double discount;  // 折扣，0-1 之间
    
    public DiscountProduct(String id, String name, double price, int stock, double discount) {
        super(id, name, price, stock);
        setDiscount(discount);
    }
    
    public void setDiscount(double discount) {
        if (discount > 0 && discount <= 1) {
            this.discount = discount;
        }
    }
    
    public double getDiscountedPrice() {
        return price * discount;
    }
    
    @Override
    public double getTotalPrice() {
        return getDiscountedPrice() * stock;
    }
    
    @Override
    public String toString() {
        return "DiscountProduct{" +
                "id='" + id + "', name='" + name + '\'' +
                ", originalPrice=" + price +
                ", discountedPrice=" + getDiscountedPrice() +
                ", discount=" + (discount * 10) + "折" +
                '}';
    }
}
```

### 1.2 订单系统

```java
class OrderItem {
    private Product product;
    private int quantity;
    
    public OrderItem(Product product, int quantity) {
        this.product = product;
        this.quantity = quantity;
    }
    
    public double getSubtotal() {
        if (product instanceof DiscountProduct) {
            return ((DiscountProduct) product).getDiscountedPrice() * quantity;
        }
        return product.getPrice() * quantity;
    }
    
    @Override
    public String toString() {
        return product.getName() + " x " + quantity + " = " + getSubtotal();
    }
}

class Order {
    private String orderId;
    private List<OrderItem> items = new ArrayList<>();
    
    public Order(String orderId) {
        this.orderId = orderId;
    }
    
    public void addItem(Product product, int quantity) {
        items.add(new OrderItem(product, quantity));
    }
    
    public double getTotal() {
        double total = 0;
        for (OrderItem item : items) {
            total += item.getSubtotal();
        }
        return total;
    }
    
    public void printOrder() {
        System.out.println("订单号：" + orderId);
        for (OrderItem item : items) {
            System.out.println("  " + item);
        }
        System.out.println("总计：" + getTotal());
    }
}
```

## 二、案例二：银行系统

### 2.1 账户类设计

```java
// 父类：银行账户
abstract class BankAccount {
    protected String accountNumber;
    protected String owner;
    protected double balance;
    
    public BankAccount(String accountNumber, String owner, double initialBalance) {
        this.accountNumber = accountNumber;
        this.owner = owner;
        this.balance = initialBalance;
    }
    
    public void deposit(double amount) {
        if (amount > 0) {
            balance += amount;
            System.out.println("存入：" + amount + "，余额：" + balance);
        }
    }
    
    public abstract boolean withdraw(double amount);
    
    public double getBalance() {
        return balance;
    }
    
    public void printInfo() {
        System.out.println("账号：" + accountNumber + "，户名：" + owner + "，余额：" + balance);
    }
}

// 子类：活期账户
class SavingsAccount extends BankAccount {
    private double interestRate;
    
    public SavingsAccount(String accountNumber, String owner, double initialBalance, 
                         double interestRate) {
        super(accountNumber, owner, initialBalance);
        this.interestRate = interestRate;
    }
    
    @Override
    public boolean withdraw(double amount) {
        if (amount > 0 && amount <= balance) {
            balance -= amount;
            System.out.println("取出：" + amount + "，余额：" + balance);
            return true;
        }
        System.out.println("取款失败");
        return false;
    }
    
    public void addInterest() {
        double interest = balance * interestRate;
        balance += interest;
        System.out.println("利息：" + interest + "，新余额：" + balance);
    }
}

// 子类：定期账户
class TimeDepositAccount extends BankAccount {
    private int termMonths;
    private double interestRate;
    private boolean isWithdrawn;
    
    public TimeDepositAccount(String accountNumber, String owner, double initialBalance,
                             int termMonths, double interestRate) {
        super(accountNumber, owner, initialBalance);
        this.termMonths = termMonths;
        this.interestRate = interestRate;
        this.isWithdrawn = false;
    }
    
    @Override
    public boolean withdraw(double amount) {
        if (isWithdrawn) {
            System.out.println("已提前支取，按活期计息");
            return false;
        }
        if (amount > balance) {
            System.out.println("余额不足");
            return false;
        }
        // 定期账户到期才能全额取款
        System.out.println("请到期后取款");
        return false;
    }
    
    public double getMaturityValue() {
        return balance * (1 + interestRate * termMonths / 12.0);
    }
}

// 子类：支票账户
class CheckingAccount extends BankAccount {
    private double overdraftLimit;
    
    public CheckingAccount(String accountNumber, String owner, double initialBalance,
                          double overdraftLimit) {
        super(accountNumber, owner, initialBalance);
        this.overdraftLimit = overdraftLimit;
    }
    
    @Override
    public boolean withdraw(double amount) {
        if (amount > 0 && amount <= balance + overdraftLimit) {
            balance -= amount;
            System.out.println("取出：" + amount + "，余额：" + balance);
            return true;
        }
        System.out.println("超出透支额度");
        return false;
    }
}
```

### 2.2 使用示例

```java
public class BankDemo {
    public static void main(String[] args) {
        // 创建账户
        BankAccount acc1 = new SavingsAccount("SA001", "张三", 10000, 0.03);
        BankAccount acc2 = new TimeDepositAccount("TD001", "李四", 50000, 12, 0.04);
        BankAccount acc3 = new CheckingAccount("CA001", "王五", 5000, 10000);
        
        // 存款
        acc1.deposit(5000);
        acc2.deposit(10000);
        acc3.deposit(2000);
        
        // 取款
        acc1.withdraw(3000);
        acc3.withdraw(12000);  // 使用透支额度
        
        // 打印信息
        acc1.printInfo();
        acc2.printInfo();
        acc3.printInfo();
        
        // 计算利息
        if (acc1 instanceof SavingsAccount) {
            ((SavingsAccount) acc1).addInterest();
        }
        
        if (acc2 instanceof TimeDepositAccount) {
            System.out.println("到期本息：" + ((TimeDepositAccount) acc2).getMaturityValue());
        }
    }
}
```

## 三、案例三：学校管理系统

### 3.1 人员类设计

```java
// 抽象父类：学校人员
abstract class SchoolPerson {
    protected String id;
    protected String name;
    protected int age;
    
    public SchoolPerson(String id, String name, int age) {
        this.id = id;
        this.name = name;
        this.age = age;
    }
    
    public abstract void introduce();
    
    @Override
    public String toString() {
        return "SchoolPerson{id='" + id + "', name='" + name + "', age=" + age + "}";
    }
}

// 子类：学生
class Student extends SchoolPerson {
    private String className;
    private double gpa;
    private List<String> courses = new ArrayList<>();
    
    public Student(String id, String name, int age, String className) {
        super(id, name, age);
        this.className = className;
        this.gpa = 0;
    }
    
    public void enrollCourse(String course) {
        courses.add(course);
        System.out.println(name + " 选修了 " + course);
    }
    
    public void setGpa(double gpa) {
        this.gpa = gpa;
    }
    
    @Override
    public void introduce() {
        System.out.println("学生：" + name + "，班级：" + className + "，GPA：" + gpa);
        System.out.print("选修课程：");
        for (String course : courses) {
            System.out.print(course + " ");
        }
        System.out.println();
    }
}

// 子类：教师
class Teacher extends SchoolPerson {
    private String department;
    private String title;
    private List<String> teachingCourses = new ArrayList<>();
    private double salary;
    
    public Teacher(String id, String name, int age, String department, String title) {
        super(id, name, age);
        this.department = department;
        this.title = title;
        this.salary = 8000;
    }
    
    public void assignCourse(String course) {
        teachingCourses.add(course);
        System.out.println(name + " 被分配教授 " + course);
    }
    
    public void promote(String newTitle) {
        this.title = newTitle;
        salary *= 1.2;
        System.out.println(name + " 晋升为 " + newTitle + "，新工资：" + salary);
    }
    
    @Override
    public void introduce() {
        System.out.println("教师：" + name + "，职称：" + title + "，部门：" + department);
        System.out.print("教授课程：");
        for (String course : teachingCourses) {
            System.out.print(course + " ");
        }
        System.out.println();
    }
}

// 子类：管理员
class Administrator extends SchoolPerson {
    private String position;
    private String office;
    private double salary;
    
    public Administrator(String id, String name, int age, String position, String office) {
        super(id, name, age);
        this.position = position;
        this.office = office;
        this.salary = 6000;
    }
    
    @Override
    public void introduce() {
        System.out.println("管理员：" + name + "，职位：" + position + "，办公室：" + office);
    }
}
```

### 3.2 使用示例

```java
public class SchoolDemo {
    public static void main(String[] args) {
        // 创建人员
        SchoolPerson p1 = new Student("S001", "张三", 18, "计算机 1 班");
        SchoolPerson p2 = new Teacher("T001", "李老师", 35, "计算机学院", "副教授");
        SchoolPerson p3 = new Administrator("A001", "王主任", 40, "教务处", "办公楼 101");
        
        // 添加到列表
        List<SchoolPerson> people = new ArrayList<>();
        people.add(p1);
        people.add(p2);
        people.add(p3);
        
        // 统一介绍
        for (SchoolPerson person : people) {
            person.introduce();
            System.out.println();
        }
        
        // 类型转换调用特有方法
        if (p1 instanceof Student) {
            Student s = (Student) p1;
            s.enrollCourse("Java 编程");
            s.enrollCourse("数据结构");
            s.setGpa(3.8);
        }
        
        if (p2 instanceof Teacher) {
            Teacher t = (Teacher) p2;
            t.assignCourse("Java 编程");
            t.assignCourse("数据结构");
            t.promote("教授");
        }
    }
}
```
